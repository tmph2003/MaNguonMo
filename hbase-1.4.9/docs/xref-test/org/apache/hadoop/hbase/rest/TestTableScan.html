255   /**
256    * An example to scan using listener in unmarshaller for XML.
257    * @throws Exception the exception
258    */
259   @Test
260   public void testScanUsingListenerUnmarshallerXML() throws Exception {
261     StringBuilder builder = new StringBuilder();
262     builder.append("/*");
263     builder.append("?");
264     builder.append(Constants.SCAN_COLUMN + "=" + COLUMN_1);
265     builder.append("&");
266     builder.append(Constants.SCAN_LIMIT + "=10");
267     Response response = client.get("/" + TABLE + builder.toString(),
268       Constants.MIMETYPE_XML);
269     assertEquals(200, response.getCode());
270     assertEquals(Constants.MIMETYPE_XML, response.getHeader("content-type"));
271     JAXBContext context = JAXBContext.newInstance(ClientSideCellSetModel.class, RowModel.class,
272       CellModel.class);
273     Unmarshaller unmarshaller = context.createUnmarshaller();
274 
275     final ClientSideCellSetModel.Listener listener = new ClientSideCellSetModel.Listener() {
276       @Override
277       public void handleRowModel(ClientSideCellSetModel helper, RowModel row) {
278         assertTrue(row.getKey() != null);
279         assertTrue(row.getCells().size() > 0);
280       }
281     };
282 
283     // install the callback on all ClientSideCellSetModel instances
284     unmarshaller.setListener(new Unmarshaller.Listener() {
285         @Override
286         public void beforeUnmarshal(Object target, Object parent) {
287             if (target instanceof ClientSideCellSetModel) {
288                 ((ClientSideCellSetModel) target).setCellSetModelListener(listener);
289             }
290         }
291 
292         @Override
293         public void afterUnmarshal(Object target, Object parent) {
294             if (target instanceof ClientSideCellSetModel) {
295                 ((ClientSideCellSetModel) target).setCellSetModelListener(null);
296             }
297         }
298     });
299 
300     // create a new XML parser
301     SAXParserFactory factory = SAXParserFactory.newInstance();
302     factory.setNamespaceAware(true);
303     XMLReader reader = factory.newSAXParser().getXMLReader();
304     reader.setContentHandler(unmarshaller.getUnmarshallerHandler());
305     assertFalse(ClientSideCellSetModel.listenerInvoked);
306     reader.parse(new InputSource(response.getStream()));
307     assertTrue(ClientSideCellSetModel.listenerInvoked);
308 
309   }

331     //Test scanning with no limit.
332     builder = new StringBuilder();
333     builder.append("/*");

335     builder.append(Constants.SCAN_COLUMN + "=" + COLUMN_2);
336     response = client.get("/" + TABLE + builder.toString(),
337       Constants.MIMETYPE_JSON);
338     assertEquals(200, response.getCode());
339     assertEquals(Constants.MIMETYPE_JSON, response.getHeader("content-type"));
340     model = mapper.readValue(response.getStream(), CellSetModel.class);
341     count = TestScannerResource.countCellSet(model);
342     assertEquals(expectedRows2, count);
343     checkRowsNotNull(model);
344 
345     //Test with start row and end row.
346     builder = new StringBuilder();
347     builder.append("/*");
348     builder.append("?");
349     builder.append(Constants.SCAN_COLUMN + "=" + COLUMN_1);
350     builder.append("&");
351     builder.append(Constants.SCAN_START_ROW + "=aaa");
352     builder.append("&");
353     builder.append(Constants.SCAN_END_ROW + "=aay");
354     response = client.get("/" + TABLE + builder.toString(),
355       Constants.MIMETYPE_JSON);
356     assertEquals(200, response.getCode());
357 
358     count = 0;
359     JsonFactory jfactory = new JsonFactory(mapper);
360     JsonParser jParser = jfactory.createJsonParser(response.getStream());
361     boolean found = false;
362     while (jParser.nextToken() != JsonToken.END_OBJECT) {
363       if(jParser.getCurrentToken() == JsonToken.START_OBJECT && found) {
364         RowModel row = jParser.readValueAs(RowModel.class);
365         assertNotNull(row.getKey());
366         for (int i = 0; i < row.getCells().size(); i++) {
367           if (count == 0) {
368             assertEquals("aaa", Bytes.toString(row.getKey()));
369           }
370           if (count == 23) {
371             assertEquals("aax", Bytes.toString(row.getKey()));
372           }
373           count++;
374         }
375         jParser.skipChildren();
376       } else {
377         found = jParser.getCurrentToken() == JsonToken.START_ARRAY;
378       }
379     }
380     assertEquals(24, count);
381   }

415   private void checkRowsNotNull(CellSetModel model) {
416     for (RowModel row: model.getRows()) {
417       assertTrue(row.getKey() != null);
418       assertTrue(row.getCells().size() > 0);
419     }

422   /**
423    * Read protobuf stream.
424    * @param inputStream the input stream
425    * @return The number of rows in the cell set model.
426    * @throws IOException Signals that an I/O exception has occurred.
427    */
428   public int readProtobufStream(InputStream inputStream) throws IOException{
429     DataInputStream stream = new DataInputStream(inputStream);
430     CellSetModel model = null;
431     int rowCount = 0;
432     try {
433       while (true) {
434         byte[] lengthBytes = new byte[2];
435         int readBytes = stream.read(lengthBytes);
436         if (readBytes == -1) {
437           break;
438         }
439         assertEquals(2, readBytes);
440         int length = Bytes.toShort(lengthBytes);
441         byte[] cellset = new byte[length];
442         stream.read(cellset);
443         model = new CellSetModel();
444         model.getObjectFromMessage(cellset);
445         checkRowsNotNull(model);
446         rowCount = rowCount + TestScannerResource.countCellSet(model);
447       }
448     } catch (EOFException exp) {
449       exp.printStackTrace();
450     } finally {
451       stream.close();
452     }
453     return rowCount;
454   }

687    * user defined operations on the row model.
688    */
689   @XmlRootElement(name = "CellSet")
690   @XmlAccessorType(XmlAccessType.FIELD)
691   public static class ClientSideCellSetModel implements Serializable {
692 
693     private static final long serialVersionUID = 1L;
694 
695     /**
696      * This list is not a real list; instead it will notify a listener whenever JAXB has
697      * unmarshalled the next row.
698      */
699     @XmlElement(name="Row")
700     private List<RowModel> row;
701 
702     static boolean listenerInvoked = false;
703 
704     /**
705      * Install a listener for row model on this object. If l is null, the listener
706      * is removed again.
707      */
708     public void setCellSetModelListener(final Listener l) {
709         row = (l == null) ? null : new ArrayList<RowModel>() {
710         private static final long serialVersionUID = 1L;
711             @Override
712             public boolean add(RowModel o) {
713                 l.handleRowModel(ClientSideCellSetModel.this, o);
714                 listenerInvoked = true;
715                 return false;
716             }
717         };
718     }
719 
720     /**
721      * This listener is invoked every time a new row model is unmarshalled.
722      */
723     public static interface Listener {
724         void handleRowModel(ClientSideCellSetModel helper, RowModel rowModel);
725     }
726   }
