66    @Test
67    public void testReplicatorBatching() throws Exception {
68      // Clear the tables
69      truncateTable(utility1, tableName);
70      truncateTable(utility2, tableName);
71  
72      // Replace the peer set up for us by the base class with a wrapper for this test
73      admin.addPeer("testReplicatorBatching",
74        new ReplicationPeerConfig().setClusterKey(utility2.getClusterKey())
75          .setReplicationEndpointImpl(ReplicationEndpointForTest.class.getName()), null);
76  
77      ReplicationEndpointForTest.setBatchCount(0);
78      ReplicationEndpointForTest.setEntriesCount(0);
79      try {
80        ReplicationEndpointForTest.pause();
81        try {
82          // Queue up a bunch of cells of size 8K. Because of RPC size limits, they will all
83          // have to be replicated separately.
84          final byte[] valueBytes = new byte[8 *1024];
85          for (int i = 0; i < NUM_ROWS; i++) {
86            htable1.put(new Put(("row"+Integer.toString(i)).getBytes())
87              .addColumn(famName, null, valueBytes)
88            );
89          }
90        } finally {
91          ReplicationEndpointForTest.resume();
92        }
93  
94        // Wait for replication to complete.
95        // We can expect 10 batches, 1 row each
96        Waiter.waitFor(conf1, 60000, new Waiter.ExplainingPredicate<Exception>() {
97          @Override
98          public boolean evaluate() throws Exception {
99            LOG.info("Count=" + ReplicationEndpointForTest.getBatchCount());
100           return ReplicationEndpointForTest.getBatchCount() >= NUM_ROWS;
101         }
102 
103         @Override
104         public String explainFailure() throws Exception {
105           return "We waited too long for expected replication of " + NUM_ROWS + " entries";
106         }
107       });
108 
109       assertEquals("We sent an incorrect number of batches", NUM_ROWS,
110         ReplicationEndpointForTest.getBatchCount());
111       assertEquals("We did not replicate enough rows", NUM_ROWS,
112         utility2.countRows(htable2));
113     } finally {
114       admin.removePeer("testReplicatorBatching");
115     }
116   }

179   public static class ReplicationEndpointForTest extends HBaseInterClusterReplicationEndpoint {
180 
181     private static AtomicInteger batchCount = new AtomicInteger(0);
182     private static int entriesCount;
183     private static final Object latch = new Object();
184     private static AtomicBoolean useLatch = new AtomicBoolean(false);
185 
186     public static void resume() {
187       useLatch.set(false);
188       synchronized (latch) {
189         latch.notifyAll();
190       }
191     }
192 
193     public static void pause() {
194       useLatch.set(true);
195     }
196 
197     public static void await() throws InterruptedException {
198       if (useLatch.get()) {
199         LOG.info("Waiting on latch");
200         synchronized(latch) {
201           latch.wait();
202         }
203         LOG.info("Waited on latch, now proceeding");
204       }
205     }
206 
207     public static int getBatchCount() {
208       return batchCount.get();
209     }
210 
211     public static void setBatchCount(int i) {
212       LOG.info("SetBatchCount=" + i + ", old=" + getBatchCount());
213       batchCount.set(i);
214     }
215 
216     public static int getEntriesCount() {
217       return entriesCount;
218     }
219 
220     public static void setEntriesCount(int i) {
221       LOG.info("SetEntriesCount=" + i);
222       entriesCount = i;
223     }

225     public class ReplicatorForTest extends Replicator {
226 
227       public ReplicatorForTest(List<Entry> entries, int ordinal) {
228         super(entries, ordinal);
229       }
230 
231       @Override
232       protected void replicateEntries(BlockingInterface rrs, final List<Entry> entries,
233           String replicationClusterId, Path baseNamespaceDir, Path hfileArchiveDir)
234           throws IOException {
235         try {
236           long size = 0;
237           for (Entry e: entries) {
238             size += e.getKey().estimatedSerializedSizeOf();
239             size += e.getEdit().estimatedSerializedSizeOf();
240           }
241           LOG.info("Replicating batch " + System.identityHashCode(entries) + " of " +
242               entries.size() + " entries with total size " + size + " bytes to " +
243               replicationClusterId);
244           super.replicateEntries(rrs, entries, replicationClusterId, baseNamespaceDir,
245             hfileArchiveDir);
246           entriesCount += entries.size();
247           int count = batchCount.incrementAndGet();
248           LOG.info("Completed replicating batch " + System.identityHashCode(entries) +
249               " count=" + count);
250         } catch (IOException e) {
251           LOG.info("Failed to replicate batch " + System.identityHashCode(entries), e);
252           throw e;
253         }
254       }
255     }
256 
257     @Override
258     public boolean replicate(ReplicateContext replicateContext) {
259       try {
260         await();
261       } catch (InterruptedException e) {
262         LOG.warn("Interrupted waiting for latch", e);
263       }
264       return super.replicate(replicateContext);
265     }
266 
267     @Override
268     protected Replicator createReplicator(List<Entry> entries, int ordinal) {
269       return new ReplicatorForTest(entries, ordinal);
270     }
271   }
