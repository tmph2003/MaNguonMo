219   @Test
220   public void testLogRoll() throws Exception {
221     long baseline = 1000;
222     long time = baseline;
223     MultiVersionConcurrencyControl mvcc = new MultiVersionConcurrencyControl();
224     KeyValue kv = new KeyValue(r1, f1, r1);
225     WALEdit edit = new WALEdit();
226     edit.add(kv);
227 
228     List<WALActionsListener> listeners = new ArrayList<WALActionsListener>();
229     listeners.add(replication);
230     final WALFactory wals = new WALFactory(utility.getConfiguration(), listeners,
231         URLEncoder.encode("regionserver:60020", "UTF8"));
232     final WAL wal = wals.getWAL(hri.getEncodedNameAsBytes(), hri.getTable().getNamespace());
233     manager.init();
234     HTableDescriptor htd = new HTableDescriptor();
235     htd.addFamily(new HColumnDescriptor(f1));
236     // Testing normal log rolling every 20
237     for(long i = 1; i < 101; i++) {
238       if(i > 1 && i % 20 == 0) {
239         wal.rollWriter();
240       }
241       LOG.info(i);
242       final long txid = wal.append(htd,
243           hri,
244           new WALKey(hri.getEncodedNameAsBytes(), test, System.currentTimeMillis(), mvcc),
245           edit,
246           true);
247       wal.sync(txid);
248     }
249 
250     // Simulate a rapid insert that's followed
251     // by a report that's still not totally complete (missing last one)
252     LOG.info(baseline + " and " + time);
253     baseline += 101;
254     time = baseline;
255     LOG.info(baseline + " and " + time);
256 
257     for (int i = 0; i < 3; i++) {
258       wal.append(htd, hri,
259           new WALKey(hri.getEncodedNameAsBytes(), test, System.currentTimeMillis(), mvcc),
260           edit,
261           true);
262     }
263     wal.sync();
264 
265     int logNumber = 0;
266     for (Map.Entry<String, SortedSet<String>> entry : manager.getWALs().get(slaveId).entrySet()) {
267       logNumber += entry.getValue().size();
268     }
269     assertEquals(6, logNumber);
270 
271     wal.rollWriter();
272 
273     manager.logPositionAndCleanOldLogs(manager.getSources().get(0).getCurrentPath(),
274         "1", 0, false, false);
275 
276     wal.append(htd, hri,
277         new WALKey(hri.getEncodedNameAsBytes(), test, System.currentTimeMillis(), mvcc),
278         edit,
279         true);
280     wal.sync();
281 
282     assertEquals(1, manager.getWALs().size());
283 
284 
285     // TODO Need a case with only 2 WALs and we only want to delete the first one
286   }

311         server.getServerName().getServerName(), s2);
312     DummyNodeFailoverWorker w3 = new DummyNodeFailoverWorker(
313         server.getServerName().getServerName(), s3);
314 
315     latch = new CountDownLatch(3);
316     // start the threads
317     w1.start();
318     w2.start();

529     final String peerId = "FakePeer";

554       // Need to wait for the ReplicationTracker to pick up the changes and notify listeners.

557     } finally {
558       conf.set("replication.replicationsource.implementation", replicationSourceImplName);
559       removePeerAndWait(peerId);

579       int globalLogQueueSizeInitial = globalSource.getSizeOfLogQueue();

582       source.enqueueLog(new Path("abc"));
583       assertEquals(1, source.getSourceMetrics().getSizeOfLogQueue());
584       assertEquals(1 + globalLogQueueSizeInitial, globalSource.getSizeOfLogQueue());

594       assertEquals(0, source.getSourceMetrics().getSizeOfLogQueue());

596     } finally {
