268   /**
269    * Tests the cyclic replication scenario of 0 -> 1 -> 2 -> 0 by adding and deleting rows to a
270    * table in each clusters and ensuring that the each of these clusters get the appropriate
271    * mutations. It also tests the grouping scenario where a cluster needs to replicate the edits
272    * originating from itself and also the edits that it received using replication from a different
273    * cluster. The scenario is explained in HBASE-9158
274    */
275   @Test(timeout = 300000)
276   public void testCyclicReplication2() throws Exception {
277     LOG.info("testCyclicReplication2");
278     int numClusters = 3;
279     Table[] htables = null;
280     try {
281       startMiniClusters(numClusters);
282       createTableOnClusters(table);
283 
284       // Test the replication scenario of 0 -> 1 -> 2 -> 0
285       addPeer("1", 0, 1);
286       addPeer("1", 1, 2);
287       addPeer("1", 2, 0);
288 
289       htables = getHTablesOnClusters(tableName);
290 
291       // put "row" and wait 'til it got around
292       putAndWait(row, famName, htables[0], htables[2]);
293       putAndWait(row1, famName, htables[1], htables[0]);
294       putAndWait(row2, famName, htables[2], htables[1]);
295 
296       deleteAndWait(row, htables[0], htables[2]);
297       deleteAndWait(row1, htables[1], htables[0]);
298       deleteAndWait(row2, htables[2], htables[1]);
299 
300       int[] expectedCounts = new int[] { 3, 3, 3 };
301       validateCounts(htables, put, expectedCounts);
302       validateCounts(htables, delete, expectedCounts);
303 
304       // Test HBASE-9158
305       disablePeer("1", 2);
306       // we now have an edit that was replicated into cluster originating from
307       // cluster 0
308       putAndWait(row3, famName, htables[0], htables[1]);
309       // now add a local edit to cluster 1
310       htables[1].put(new Put(row4).add(famName, row4, row4));
311       // re-enable replication from cluster 2 to cluster 0
312       enablePeer("1", 2);
313       // without HBASE-9158 the edit for row4 would have been marked with
314       // cluster 0's id
315       // and hence not replicated to cluster 0
316       wait(row4, htables[0], false);
317     } finally {
318       close(htables);
319       shutDownMiniClusters();
320     }
321   }

502   @SuppressWarnings("resource")
503   private void startMiniClusters(int numClusters) throws Exception {
504     Random random = new Random();
505     utilities = new HBaseTestingUtility[numClusters];
506     configurations = new Configuration[numClusters];
507     for (int i = 0; i < numClusters; i++) {
508       Configuration conf = new Configuration(baseConfiguration);
509       conf.set(HConstants.ZOOKEEPER_ZNODE_PARENT, "/" + i + random.nextInt());
510       HBaseTestingUtility utility = new HBaseTestingUtility(conf);
511       if (i == 0) {
512         utility.startMiniZKCluster();
513         miniZK = utility.getZkCluster();
514       } else {
515         utility.setZkCluster(miniZK);
516       }
517       utility.startMiniCluster();
518       utilities[i] = utility;
519       configurations[i] = conf;
520       new ZooKeeperWatcher(conf, "cluster" + i, null, true);
521     }
522   }
523 
524   private void shutDownMiniClusters() throws Exception {
525     int numClusters = utilities.length;
526     for (int i = numClusters - 1; i >= 0; i--) {
527       if (utilities[i] != null) {
528         utilities[i].shutdownMiniCluster();
529       }
530     }
531     miniZK.shutdown();
532   }

681   private void wait(int slaveNumber, Table target, int expectedCount)
682       throws IOException, InterruptedException {
683     int count = 0;
684     for (int i = 0; i < NB_RETRIES; i++) {
685       if (i == NB_RETRIES - 1) {
686         fail("Waited too much time for bulkloaded data replication. Current count=" + count
687             + ", expected count=" + expectedCount);
688       }
689       count = utilities[slaveNumber].countRows(target);
690       if (count != expectedCount) {
691         LOG.info("Waiting more time for bulkloaded data replication.");
692         Thread.sleep(SLEEP_TIME);
693       } else {
694         break;
695       }
696     }
697   }

699   private void wait(byte[] row, Table target, boolean isDeleted) throws Exception {
700     Get get = new Get(row);
701     for (int i = 0; i < NB_RETRIES; i++) {
702       if (i == NB_RETRIES - 1) {
703         fail("Waited too much time for replication. Row:" + Bytes.toString(row)
704             + ". IsDeleteReplication:" + isDeleted);
705       }
706       Result res = target.get(get);
707       boolean sleep = isDeleted ? res.size() > 0 : res.size() == 0;
708       if (sleep) {
709         LOG.info("Waiting for more time for replication. Row:"
710             + Bytes.toString(row) + ". IsDeleteReplication:" + isDeleted);
711         Thread.sleep(SLEEP_TIME);
712       } else {
713         if (!isDeleted) {
714           assertArrayEquals(res.value(), row);
715         }
716         LOG.info("Obtained row:"
717             + Bytes.toString(row) + ". IsDeleteReplication:" + isDeleted);
718         break;
719       }
720     }
721   }

723   private void rollWALAndWait(final HBaseTestingUtility utility, final TableName table,
724       final byte[] row) throws IOException {
725     final Admin admin = utility.getHBaseAdmin();
726     final MiniHBaseCluster cluster = utility.getMiniHBaseCluster();
727 
728     // find the region that corresponds to the given row.
729     HRegion region = null;
730     for (HRegion candidate : cluster.getRegions(table)) {
731       if (HRegion.rowIsInRange(candidate.getRegionInfo(), row)) {
732         region = candidate;
733         break;
734       }
735     }
736     assertNotNull("Couldn't find the region for row '" + Arrays.toString(row) + "'", region);
737 
738     final CountDownLatch latch = new CountDownLatch(1);
739 
740     // listen for successful log rolls
741     final WALActionsListener listener = new WALActionsListener.Base() {
742           @Override
743           public void postLogRoll(final Path oldPath, final Path newPath) throws IOException {
744             latch.countDown();
745           }
746         };
747     region.getWAL().registerWALActionsListener(listener);
748 
749     // request a roll
750     admin.rollWALWriter(cluster.getServerHoldingRegion(region.getTableDesc().getTableName(),
751       region.getRegionInfo().getRegionName()));
752 
753     // wait
754     try {
755       latch.await();
756     } catch (InterruptedException exception) {
757       LOG.warn("Interrupted while waiting for the wal of '" + region + "' to roll. If later " +
758           "replication tests fail, it's probably because we should still be waiting.");
759       Thread.currentThread().interrupt();
760     }
761     region.getWAL().unregisterWALActionsListener(listener);
762   }

764   /**
765    * Use a coprocessor to count puts and deletes. as KVs would be replicated back with the same
766    * timestamp there is otherwise no way to count them.
767    */
768   public static class CoprocessorCounter extends BaseRegionObserver {
769     private int nCount = 0;
770     private int nDelete = 0;
771 
772     @Override
773     public void prePut(final ObserverContext<RegionCoprocessorEnvironment> e, final Put put,
774         final WALEdit edit, final Durability durability) throws IOException {
775       nCount++;
776     }
777 
778     @Override
779     public void postDelete(final ObserverContext<RegionCoprocessorEnvironment> c,
780         final Delete delete, final WALEdit edit, final Durability durability) throws IOException {
781       nDelete++;
782     }
783 
784     @Override
785     public void preGetOp(final ObserverContext<RegionCoprocessorEnvironment> c,
786         final Get get, final List<Cell> result) throws IOException {
787       if (get.getAttribute("count") != null) {
788         result.clear();
789         // order is important!
790         result.add(new KeyValue(count, count, delete, Bytes.toBytes(nDelete)));
791         result.add(new KeyValue(count, count, put, Bytes.toBytes(nCount)));
792         c.bypass();
793       }
794     }
795   }
