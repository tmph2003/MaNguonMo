83    @BeforeClass
84    public static void beforeClass() throws Exception {
85      UTIL.getConfiguration().setBoolean("hbase.thrift.readonly", true);
86      UTIL.getConfiguration().set("hbase.client.retries.number", "3");
87      UTIL.startMiniCluster();
88      Admin admin = UTIL.getHBaseAdmin();
89      HTableDescriptor tableDescriptor = new HTableDescriptor(TableName.valueOf(tableAname));
90      for (HColumnDescriptor family : families) {
91        tableDescriptor.addFamily(family);
92      }
93      admin.createTable(tableDescriptor);
94      admin.close();
