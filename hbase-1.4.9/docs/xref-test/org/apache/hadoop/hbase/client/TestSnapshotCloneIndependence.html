93    static void setupConf(Configuration conf) {
94      // Up the handlers; this test needs more than usual.
95      conf.setInt(HConstants.REGION_SERVER_HIGH_PRIORITY_HANDLER_COUNT, 15);
96      // enable snapshot support
97      conf.setBoolean(SnapshotManager.HBASE_SNAPSHOT_ENABLED, true);
98      // disable the ui
99      conf.setInt("hbase.regionsever.info.port", -1);
100     conf.setInt("hbase.master.info.port", -1);
101     // change the flush size to a small amount, regulating number of store files
102     conf.setInt("hbase.hregion.memstore.flush.size", 25000);
103     // so make sure we get a compaction when doing a load, but keep around
104     // some files in the store
105     conf.setInt("hbase.hstore.compaction.min", 10);
106     conf.setInt("hbase.hstore.compactionThreshold", 10);
107     // block writes if we get to 12 store files
108     conf.setInt("hbase.hstore.blockingStoreFiles", 12);
109     conf.setInt("hbase.regionserver.msginterval", 100);
110     conf.setBoolean("hbase.master.enabletable.roundrobin", true);
111     // Avoid potentially aggressive splitting which would cause snapshot to fail
112     conf.set(HConstants.HBASE_REGION_SPLIT_POLICY_KEY,
113         ConstantSizeRegionSplitPolicy.class.getName());
114     // Execute cleaner frequently to induce failures
115     conf.setInt("hbase.master.cleaner.interval", CLEANER_INTERVAL);
116     conf.setInt("hbase.master.hfilecleaner.plugins.snapshot.period", CLEANER_INTERVAL);
117     // Effectively disable TimeToLiveHFileCleaner. Don't want to fully disable it because that
118     // will even trigger races between creating the directory containing back references and
119     // the back reference itself.
120     conf.setInt("hbase.master.hfilecleaner.ttl", CLEANER_INTERVAL);
121   }

229   private static void waitOnSplit(Connection c, final Table t, int originalCount) throws Exception {
230     for (int i = 0; i < 200; i++) {
231       Threads.sleepWithoutInterrupt(500);
232       try (RegionLocator locator = c.getRegionLocator(t.getName())) {
233         if (locator.getAllRegionLocations().size() > originalCount) {
234           return;
235         }
236       }
237     }
238     throw new Exception("Split did not increase the number of regions");
239   }

241   /**
242    * Takes the snapshot of originalTable and clones the snapshot to another tables.
243    * If {@code online} is false, the original table is disabled during taking snapshot, so also
244    * enables it again.
245    * @param online - Whether the table is online or not during the snapshot
246    */
247   private void createAndCloneSnapshot(boolean online) throws Exception {
248     SnapshotTestingUtils.createSnapshotAndValidate(admin, originalTableName, TEST_FAM_STR,
249       snapshotNameAsString, rootDir, fs, online);
250 
251     // If offline, enable the table disabled by snapshot testing util.
252     if (!online) {
253       admin.enableTable(originalTableName);
254       UTIL.waitTableAvailable(originalTableName);
255     }
256 
257     admin.cloneSnapshot(snapshotName, cloneTableName);
258     UTIL.waitUntilAllRegionsAssigned(cloneTableName);
259   }

261   /**
262    * Verify that adding data to original table or clone table doesn't affect other table.
263    */
264   private void runTestSnapshotAppendIndependent() throws Exception {
265     try (Table clonedTable = UTIL.getConnection().getTable(cloneTableName)) {
266       final int clonedTableRowCount = countRows(clonedTable);
267 
268       Assert.assertEquals(
269         "The line counts of original and cloned tables do not match after clone. ",
270         countOriginalTable, clonedTableRowCount);
271 
272       // Attempt to add data to the test
273       Put p = new Put(Bytes.toBytes("new-row-" + System.currentTimeMillis()));
274       p.addColumn(TEST_FAM, Bytes.toBytes("someQualifier"), Bytes.toBytes("someString"));
275       originalTable.put(p);
276 
277       // Verify that the new row is not in the restored table
278       Assert.assertEquals("The row count of the original table was not modified by the put",
279         countOriginalTable + 1, countRows(originalTable));
280       Assert.assertEquals(
281         "The row count of the cloned table changed as a result of addition to the original",
282         clonedTableRowCount, countRows(clonedTable));
283 
284       Put p2 = new Put(Bytes.toBytes("new-row-" + System.currentTimeMillis()));
285       p2.addColumn(TEST_FAM, Bytes.toBytes("someQualifier"), Bytes.toBytes("someString"));
286       clonedTable.put(p2);
287 
288       // Verify that the row is not added to the original table.
289       Assert.assertEquals(
290         "The row count of the original table was modified by the put to the clone",
291         countOriginalTable + 1, countRows(originalTable));
292       Assert.assertEquals("The row count of the cloned table was not modified by the put",
293         clonedTableRowCount + 1, countRows(clonedTable));
294     }

297   /**
298    * Do a split, and verify that this only affects one table
299    */
300   private void runTestRegionOperationsIndependent() throws Exception {
301     // Verify that region information is the same pre-split
302     ((ClusterConnection) UTIL.getConnection()).clearRegionCache();
303     List<HRegionInfo> originalTableHRegions = admin.getTableRegions(originalTableName);
304 
305     final int originalRegionCount = originalTableHRegions.size();
306     final int cloneTableRegionCount = admin.getTableRegions(cloneTableName).size();
307     Assert.assertEquals(
308       "The number of regions in the cloned table is different than in the original table.",
309       originalRegionCount, cloneTableRegionCount);
310 
311     // Split a region on the parent table
312     admin.splitRegion(originalTableHRegions.get(0).getRegionName());
313     waitOnSplit(UTIL.getConnection(), originalTable, originalRegionCount);
314 
315     // Verify that the cloned table region is not split
316     final int cloneTableRegionCount2 = admin.getTableRegions(cloneTableName).size();
317     Assert.assertEquals(
318       "The number of regions in the cloned table changed though none of its regions were split.",
319       cloneTableRegionCount, cloneTableRegionCount2);
320   }

322   /**
323    * Add metadata, and verify that this only affects one table
324    */
325   private void runTestSnapshotMetadataChangesIndependent() throws Exception {
326     // Add a new column family to the original table
327     byte[] TEST_FAM_2 = Bytes.toBytes("fam2");
328     HColumnDescriptor hcd = new HColumnDescriptor(TEST_FAM_2);
329 
330     admin.disableTable(originalTableName);
331     admin.addColumn(originalTableName, hcd);
332 
333     // Verify that it is not in the snapshot
334     admin.enableTable(originalTableName);
335     UTIL.waitTableAvailable(originalTableName);
336 
337     // get a description of the cloned table
338     // get a list of its families
339     // assert that the family is there
340     HTableDescriptor originalTableDescriptor = originalTable.getTableDescriptor();
341     HTableDescriptor clonedTableDescriptor = admin.getTableDescriptor(cloneTableName);
342 
343     Assert.assertTrue("The original family was not found. There is something wrong. ",
344       originalTableDescriptor.hasFamily(TEST_FAM));
345     Assert.assertTrue("The original family was not found in the clone. There is something wrong. ",
346       clonedTableDescriptor.hasFamily(TEST_FAM));
347 
348     Assert.assertTrue("The new family was not found. ",
349       originalTableDescriptor.hasFamily(TEST_FAM_2));
350     Assert.assertTrue("The new family was not found. ",
351       !clonedTableDescriptor.hasFamily(TEST_FAM_2));
352   }

354   /**
355    * Verify that deleting the snapshot does not affect either table.
356    */
357   private void runTestSnapshotDeleteIndependent() throws Exception {
358     // Ensure the original table does not reference the HFiles anymore
359     admin.majorCompact(originalTableName);
360 
361     // Deleting the snapshot used to break the cloned table by deleting in-use HFiles
362     admin.deleteSnapshot(snapshotName);
363 
364     // Wait for cleaner run and DFS heartbeats so that anything that is deletable is fully deleted
365     do {
366       Thread.sleep(5000);
367     } while (!admin.listSnapshots(snapshotNameAsString).isEmpty());
368 
369     try (Table original = UTIL.getConnection().getTable(originalTableName)) {
370       try (Table clonedTable = UTIL.getConnection().getTable(cloneTableName)) {
371         // Verify that all regions of both tables are readable
372         final int origTableRowCount = countRows(original);
373         final int clonedTableRowCount = countRows(clonedTable);
374         Assert.assertEquals(origTableRowCount, clonedTableRowCount);
375       }
376     }
377   }
