280               Set<PayloadCarryingServerCallable> callsInProgress) {
281         SingleServerRequestRunnable rq = new SingleServerRequestRunnable(
282                 multiAction, numAttempt, server, callsInProgress);
283         List<Long> heapCount = heapSizesByServer.get(server);
284         if (heapCount == null) {
285           heapCount = new ArrayList<>();
286           heapSizesByServer.put(server, heapCount);
287         }
288         heapCount.add(heapSizeOf(multiAction));
289         return rq;
290       }
291 
292       long heapSizeOf(MultiAction<Row> multiAction) {
293         long sum = 0;
294         for (List<Action<Row>> actions : multiAction.actions.values()) {
295           for (Action action : actions) {
296             Row row = action.getAction();
297             if (row instanceof Mutation) {
298               sum += ((Mutation) row).heapSize();
299             }
300           }
301         }
302         return sum;
303       }

387     @Override
388     protected RpcRetryingCaller<MultiResponse> createCaller(
389         PayloadCarryingServerCallable payloadCallable, int rpcTimeout) {
390       MultiServerCallable<Row> callable = (MultiServerCallable) payloadCallable;
391       final MultiResponse mr = createMultiResponse(
392           callable.getMulti(), nbMultiResponse, nbActions, new ResponseGenerator() {
393             @Override
394             public void addResponse(MultiResponse mr, byte[] regionName, Action<Row> a) {
395               if (failures.contains(regionName)) {
396                 mr.add(regionName, a.getOriginalIndex(), failure);
397               } else {
398                 boolean isStale = !RegionReplicaUtil.isDefaultReplica(a.getReplicaId());
399                 mr.add(regionName, a.getOriginalIndex(),
400                     Result.create(new Cell[0], null, isStale));
401               }
402             }
403           });
404       // Currently AsyncProcess either sends all-replica, or all-primary request.
405       final boolean isDefault = RegionReplicaUtil.isDefaultReplica(
406           callable.getMulti().actions.values().iterator().next().iterator().next().getReplicaId());
407       final ServerName server = ((MultiServerCallable<?>)callable).getServerName();
408       String debugMsg = "Call to " + server + ", primary=" + isDefault + " with "
409           + callable.getMulti().actions.size() + " entries: ";
410       for (byte[] region : callable.getMulti().actions.keySet()) {
411         debugMsg += "[" + Bytes.toStringBinary(region) + "], ";
412       }
413       LOG.debug(debugMsg);
414       if (!isDefault) {
415         replicaCalls.incrementAndGet();
416       }
417 
418       return new RpcRetryingCaller<MultiResponse>(100, 500, 10, 9) {
419         @Override
420         public MultiResponse callWithoutRetries(RetryingCallable<MultiResponse> callable,
421                                                 int callTimeout)
422         throws IOException, RuntimeException {
423           long sleep = -1;
424           if (isDefault) {
425             Long customSleep = customPrimarySleepMs.get(server);
426             sleep = (customSleep == null ? primarySleepMs : customSleep.longValue());
427           } else {
428             sleep = replicaSleepMs;
429           }
430           if (sleep != 0) {
431             try {
432               Thread.sleep(sleep);
433             } catch (InterruptedException e) {
434             }
435           }
436           return mr;
437         }
438       };
439     }

442   static MultiResponse createMultiResponse(final MultiAction<Row> multi,
443       AtomicInteger nbMultiResponse, AtomicInteger nbActions, ResponseGenerator gen) {
444     final MultiResponse mr = new MultiResponse();
445     nbMultiResponse.incrementAndGet();
446     for (Map.Entry<byte[], List<Action<Row>>> entry : multi.actions.entrySet()) {
447       byte[] regionName = entry.getKey();
448       for (Action<Row> a : entry.getValue()) {
449         nbActions.incrementAndGet();
450         gen.addResponse(mr, regionName, a);
451       }
452     }
453     return mr;
454   }

657   private void doSubmitRequest(long maxHeapSizePerRequest, long putsHeapSize) throws Exception {
658     ClusterConnection conn = createHConnection();
659     final long defaultHeapSizePerRequest = conn.getConfiguration().getLong(AsyncProcess.HBASE_CLIENT_MAX_PERREQUEST_HEAPSIZE,
660       AsyncProcess.DEFAULT_HBASE_CLIENT_MAX_PERREQUEST_HEAPSIZE);
661     conn.getConfiguration().setLong(AsyncProcess.HBASE_CLIENT_MAX_PERREQUEST_HEAPSIZE, maxHeapSizePerRequest);
662     BufferedMutatorParams bufferParam = new BufferedMutatorParams(DUMMY_TABLE);
663 
664     // sn has two regions
665     long putSizeSN = 0;
666     long putSizeSN2 = 0;
667     List<Put> puts = new ArrayList<>();
668     while ((putSizeSN + putSizeSN2) <= putsHeapSize) {
669       Put put1 = new Put(DUMMY_BYTES_1);
670       put1.addColumn(DUMMY_BYTES_1, DUMMY_BYTES_1, DUMMY_BYTES_1);
671       Put put2 = new Put(DUMMY_BYTES_2);
672       put2.addColumn(DUMMY_BYTES_2, DUMMY_BYTES_2, DUMMY_BYTES_2);
673       Put put3 = new Put(DUMMY_BYTES_3);
674       put3.addColumn(DUMMY_BYTES_3, DUMMY_BYTES_3, DUMMY_BYTES_3);
675       putSizeSN += (put1.heapSize() + put2.heapSize());
676       putSizeSN2 += put3.heapSize();
677       puts.add(put1);
678       puts.add(put2);
679       puts.add(put3);
680     }
681 
682     int minCountSnRequest = (int) calculateRequestCount(putSizeSN, maxHeapSizePerRequest);
683     int minCountSn2Request = (int) calculateRequestCount(putSizeSN2, maxHeapSizePerRequest);
684     LOG.info("Total put count:" + puts.size() + ", putSizeSN:"+ putSizeSN + ", putSizeSN2:" + putSizeSN2
685       + ", maxHeapSizePerRequest:" + maxHeapSizePerRequest
686       + ", minCountSnRequest:" + minCountSnRequest
687       + ", minCountSn2Request:" + minCountSn2Request);
688     try (HTable ht = new HTable(conn, bufferParam)) {
689       MyAsyncProcess ap = new MyAsyncProcess(conn, conf, true);
690       ht.mutator.ap = ap;
691 
692       Assert.assertEquals(0L, ht.mutator.currentWriteBufferSize.get());
693       ht.put(puts);
694       List<AsyncRequestFuture> reqs = ap.allReqs;
695 
696       int actualSnReqCount = 0;
697       int actualSn2ReqCount = 0;
698       for (AsyncRequestFuture req : reqs) {
699         if (!(req instanceof AsyncRequestFutureImpl)) {
700           continue;
701         }
702         MyAsyncProcess.MyAsyncRequestFutureImpl ars = (MyAsyncProcess.MyAsyncRequestFutureImpl) req;
703         if (ars.getRequestHeapSize().containsKey(sn)) {
704           ++actualSnReqCount;
705         }
706         if (ars.getRequestHeapSize().containsKey(sn2)) {
707           ++actualSn2ReqCount;
708         }
709       }
710       // If the server is busy, the actual count may be incremented.
711       assertEquals(true, minCountSnRequest <= actualSnReqCount);
712       assertEquals(true, minCountSn2Request <= actualSn2ReqCount);
713       Map<ServerName, Long> sizePerServers = new HashMap<>();
714       for (AsyncRequestFuture req : reqs) {
715         if (!(req instanceof AsyncRequestFutureImpl)) {
716           continue;
717         }
718         MyAsyncProcess.MyAsyncRequestFutureImpl ars = (MyAsyncProcess.MyAsyncRequestFutureImpl) req;
719         Map<ServerName, List<Long>> requestHeapSize = ars.getRequestHeapSize();
720         for (Map.Entry<ServerName, List<Long>> entry : requestHeapSize.entrySet()) {
721           long sum = 0;
722           for (long size : entry.getValue()) {
723             assertEquals(true, size <= maxHeapSizePerRequest);
724             sum += size;
725           }
726           assertEquals(true, sum <= maxHeapSizePerRequest);
727           long value = sizePerServers.containsKey(entry.getKey()) ? sizePerServers.get(entry.getKey()) : 0L;
728           sizePerServers.put(entry.getKey(), value + sum);
729         }
730       }
731       assertEquals(true, sizePerServers.containsKey(sn));
732       assertEquals(true, sizePerServers.containsKey(sn2));
733       assertEquals(false, sizePerServers.containsKey(sn3));
734       assertEquals(putSizeSN, (long) sizePerServers.get(sn));
735       assertEquals(putSizeSN2, (long) sizePerServers.get(sn2));
736     }
737     // restore config.
738     conn.getConfiguration().setLong(AsyncProcess.HBASE_CLIENT_MAX_PERREQUEST_HEAPSIZE, defaultHeapSizePerRequest);
739   }
