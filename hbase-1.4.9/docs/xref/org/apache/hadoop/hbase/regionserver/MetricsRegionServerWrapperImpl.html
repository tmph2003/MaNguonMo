93    public MetricsRegionServerWrapperImpl(final HRegionServer regionServer) {
94      this.regionServer = regionServer;
95      initBlockCache();
96  
97      this.period =
98          regionServer.conf.getLong(HConstants.REGIONSERVER_METRICS_PERIOD,
99            HConstants.DEFAULT_REGIONSERVER_METRICS_PERIOD);
100 
101     this.executor = CompatibilitySingletonFactory.getInstance(MetricsExecutor.class).getExecutor();
102     this.runnable = new RegionServerMetricsWrapperRunnable();
103     this.executor.scheduleWithFixedDelay(this.runnable, this.period, this.period,
104       TimeUnit.MILLISECONDS);
105 
106     if (LOG.isInfoEnabled()) {
107       LOG.info("Computing regionserver metrics every " + this.period + " milliseconds");
108     }
109   }

111   /**
112    * It's possible that due to threading the block cache could not be initialized
113    * yet (testing multiple region servers in one jvm).  So we need to try and initialize
114    * the blockCache and cacheStats reference multiple times until we succeed.
115    */
116   private synchronized  void initBlockCache() {
117     CacheConfig cacheConfig = this.regionServer.cacheConfig;
118     if (cacheConfig != null) {
119       l1Stats = cacheConfig.getL1Stats();
120       l2Stats = cacheConfig.getL2Stats();
121       if (this.blockCache == null) {
122         this.blockCache = cacheConfig.getBlockCache();
123       }
124     }
125 
126     if (this.blockCache != null && this.cacheStats == null) {
127       this.cacheStats = blockCache.getStats();
128     }
129   }

569   /**
570    * This is the runnable that will be executed on the executor every PERIOD number of seconds
571    * It will take metrics/numbers from all of the regions and use them to compute point in
572    * time metrics.
573    */
574   public class RegionServerMetricsWrapperRunnable implements Runnable {
575 
576     private long lastRan = 0;
577     private long lastRequestCount = 0;
578 
579     @Override
580     synchronized public void run() {
581       try {
582         initBlockCache();
583 
584         HDFSBlocksDistribution hdfsBlocksDistribution = new HDFSBlocksDistribution();
585         HDFSBlocksDistribution hdfsBlocksDistributionSecondaryRegions =
586             new HDFSBlocksDistribution();
587 
588         long tempNumStores = 0;
589         long tempNumStoreFiles = 0;
590         long tempMemstoreSize = 0;
591         long tempStoreFileSize = 0;
592         long tempMaxStoreFileAge = 0;
593         long tempNumReferenceFiles = 0;
594         long avgAgeNumerator = 0;
595         long numHFiles = 0;
596         long tempMinStoreFileAge = Long.MAX_VALUE;
597         long tempReadRequestsCount = 0;
598         long tempWriteRequestsCount = 0;
599         long tempCheckAndMutateChecksFailed = 0;
600         long tempCheckAndMutateChecksPassed = 0;
601         long tempStorefileIndexSize = 0;
602         long tempTotalStaticIndexSize = 0;
603         long tempTotalStaticBloomSize = 0;
604         long tempNumMutationsWithoutWAL = 0;
605         long tempDataInMemoryWithoutWAL = 0;
606         double tempPercentFileLocal = 0;
607         double tempPercentFileLocalSecondaryRegions = 0;
608         long tempFlushedCellsCount = 0;
609         long tempCompactedCellsCount = 0;
610         long tempMajorCompactedCellsCount = 0;
611         long tempFlushedCellsSize = 0;
612         long tempCompactedCellsSize = 0;
613         long tempMajorCompactedCellsSize = 0;
614         long tempBlockedRequestsCount = 0L;
615 
616         int regionCount = 0;
617         for (Region r : regionServer.getOnlineRegionsLocalContext()) {
618           tempNumMutationsWithoutWAL += r.getNumMutationsWithoutWAL();
619           tempDataInMemoryWithoutWAL += r.getDataInMemoryWithoutWAL();
620           tempReadRequestsCount += r.getReadRequestsCount();
621           tempWriteRequestsCount += r.getWriteRequestsCount();
622           tempCheckAndMutateChecksFailed += r.getCheckAndMutateChecksFailed();
623           tempCheckAndMutateChecksPassed += r.getCheckAndMutateChecksPassed();
624           tempBlockedRequestsCount += r.getBlockedRequestsCount();
625           List<Store> storeList = r.getStores();
626           tempNumStores += storeList.size();
627           for (Store store : storeList) {
628             tempNumStoreFiles += store.getStorefilesCount();
629             tempMemstoreSize += store.getMemStoreSize();
630             tempStoreFileSize += store.getStorefilesSize();
631 
632             long storeMaxStoreFileAge = store.getMaxStoreFileAge();
633             tempMaxStoreFileAge = (storeMaxStoreFileAge > tempMaxStoreFileAge) ?
634               storeMaxStoreFileAge : tempMaxStoreFileAge;
635 
636             long storeMinStoreFileAge = store.getMinStoreFileAge();
637             tempMinStoreFileAge = (storeMinStoreFileAge < tempMinStoreFileAge) ?
638               storeMinStoreFileAge : tempMinStoreFileAge;
639 
640             long storeHFiles = store.getNumHFiles();
641             avgAgeNumerator += store.getAvgStoreFileAge() * storeHFiles;
642             numHFiles += storeHFiles;
643             tempNumReferenceFiles += store.getNumReferenceFiles();
644 
645             tempStorefileIndexSize += store.getStorefilesIndexSize();
646             tempTotalStaticBloomSize += store.getTotalStaticBloomSize();
647             tempTotalStaticIndexSize += store.getTotalStaticIndexSize();
648             tempFlushedCellsCount += store.getFlushedCellsCount();
649             tempCompactedCellsCount += store.getCompactedCellsCount();
650             tempMajorCompactedCellsCount += store.getMajorCompactedCellsCount();
651             tempFlushedCellsSize += store.getFlushedCellsSize();
652             tempCompactedCellsSize += store.getCompactedCellsSize();
653             tempMajorCompactedCellsSize += store.getMajorCompactedCellsSize();
654           }
655 
656           HDFSBlocksDistribution distro = r.getHDFSBlocksDistribution();
657           hdfsBlocksDistribution.add(distro);
658           if (r.getRegionInfo().getReplicaId() != HRegionInfo.DEFAULT_REPLICA_ID) {
659             hdfsBlocksDistributionSecondaryRegions.add(distro);
660           }
661           regionCount++;
662         }
663 
664         float localityIndex =
665             hdfsBlocksDistribution
666                 .getBlockLocalityIndex(regionServer.getServerName().getHostname());
667         tempPercentFileLocal = Double.isNaN(tempBlockedRequestsCount) ? 0 : (localityIndex * 100);
668 
669         float localityIndexSecondaryRegions =
670             hdfsBlocksDistributionSecondaryRegions.getBlockLocalityIndex(regionServer
671                 .getServerName().getHostname());
672         tempPercentFileLocalSecondaryRegions = Double
673             .isNaN(localityIndexSecondaryRegions) ? 0 : (localityIndexSecondaryRegions * 100);
674 
675         // Compute the number of requests per second
676         long currentTime = EnvironmentEdgeManager.currentTime();
677 
678         // assume that it took PERIOD seconds to start the executor.
679         // this is a guess but it's a pretty good one.
680         if (lastRan == 0) {
681           lastRan = currentTime - period;
682         }
683 
684         // If we've time traveled keep the last requests per second.
685         if ((currentTime - lastRan) > 0) {
686           long currentRequestCount = getTotalRequestCount();
687           requestsPerSecond =
688               (currentRequestCount - lastRequestCount) / ((currentTime - lastRan) / 1000.0);
689           lastRequestCount = currentRequestCount;
690         }
691         lastRan = currentTime;
692 
693         WALProvider provider = regionServer.walFactory.getWALProvider();
694         WALProvider metaProvider = regionServer.walFactory.getMetaWALProvider();
695         numWALFiles =
696             (provider == null ? 0 : provider.getNumLogFiles())
697                 + (metaProvider == null ? 0 : metaProvider.getNumLogFiles());
698         walFileSize =
699             (provider == null ? 0 : provider.getLogFileSize())
700                 + (provider == null ? 0 : provider.getLogFileSize());
701         // Copy over computed values so that no thread sees half computed values.
702         numStores = tempNumStores;
703         numStoreFiles = tempNumStoreFiles;
704         memstoreSize = tempMemstoreSize;
705         storeFileSize = tempStoreFileSize;
706         maxStoreFileAge = tempMaxStoreFileAge;
707         if (regionCount > 0) {
708           averageRegionSize = (memstoreSize + storeFileSize) / regionCount;
709         }
710         if (tempMinStoreFileAge != Long.MAX_VALUE) {
711           minStoreFileAge = tempMinStoreFileAge;
712         }
713 
714         if (numHFiles != 0) {
715           avgStoreFileAge = avgAgeNumerator / numHFiles;
716         }
717 
718         numReferenceFiles= tempNumReferenceFiles;
719         readRequestsCount = tempReadRequestsCount;
720         writeRequestsCount = tempWriteRequestsCount;
721         checkAndMutateChecksFailed = tempCheckAndMutateChecksFailed;
722         checkAndMutateChecksPassed = tempCheckAndMutateChecksPassed;
723         storefileIndexSize = tempStorefileIndexSize;
724         totalStaticIndexSize = tempTotalStaticIndexSize;
725         totalStaticBloomSize = tempTotalStaticBloomSize;
726         numMutationsWithoutWAL = tempNumMutationsWithoutWAL;
727         dataInMemoryWithoutWAL = tempDataInMemoryWithoutWAL;
728         percentFileLocal = tempPercentFileLocal;
729         percentFileLocalSecondaryRegions = tempPercentFileLocalSecondaryRegions;
730         flushedCellsCount = tempFlushedCellsCount;
731         compactedCellsCount = tempCompactedCellsCount;
732         majorCompactedCellsCount = tempMajorCompactedCellsCount;
733         flushedCellsSize = tempFlushedCellsSize;
734         compactedCellsSize = tempCompactedCellsSize;
735         majorCompactedCellsSize = tempMajorCompactedCellsSize;
736         blockedRequestsCount = tempBlockedRequestsCount;
737       } catch (Throwable e) {
738         LOG.warn("Caught exception! Will suppress and retry.", e);
739       }
740     }
741   }
