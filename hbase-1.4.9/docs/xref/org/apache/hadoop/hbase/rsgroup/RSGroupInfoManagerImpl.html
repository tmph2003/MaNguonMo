161   public synchronized void addRSGroup(RSGroupInfo rsGroupInfo) throws IOException {
162     checkGroupName(rsGroupInfo.getName());
163     if (rsGroupMap.get(rsGroupInfo.getName()) != null ||
164         rsGroupInfo.getName().equals(RSGroupInfo.DEFAULT_GROUP)) {
165       throw new DoNotRetryIOException("Group already exists: "+ rsGroupInfo.getName());
166     }
167     Map<String, RSGroupInfo> newGroupMap = Maps.newHashMap(rsGroupMap);
168     newGroupMap.put(rsGroupInfo.getName(), rsGroupInfo);
169     flushConfig(newGroupMap);
170   }

172   @Override
173   public synchronized Set<Address> moveServers(Set<Address> servers, String srcGroup,
174                                           String dstGroup) throws IOException {
175     if (servers == null) {
176       throw new ConstraintException("The list of servers to move cannot be null.");
177     }
178     Set<Address> movedServers = Sets.newHashSet();
179     if (!rsGroupMap.containsKey(srcGroup)) {
180       throw new DoNotRetryIOException("Group "+srcGroup+" does not exist");
181     }
182     if (!rsGroupMap.containsKey(dstGroup)) {
183       throw new DoNotRetryIOException("Group "+dstGroup+" does not exist");
184     }
185 
186     RSGroupInfo src = new RSGroupInfo(getRSGroup(srcGroup));
187     RSGroupInfo dst = new RSGroupInfo(getRSGroup(dstGroup));
188     for(Address el: servers) {
189       if (src.removeServer(el)) {
190         movedServers.add(el);
191       }
192       dst.addServer(el);
193     }
194 
195     Map<String,RSGroupInfo> newGroupMap = Maps.newHashMap(rsGroupMap);
196     newGroupMap.put(src.getName(), src);
197     newGroupMap.put(dst.getName(), dst);
198 
199     flushConfig(newGroupMap);
200     return movedServers;
201   }

242   @Override
243   public synchronized void moveTables(
244       Set<TableName> tableNames, String groupName) throws IOException {
245     if (groupName != null && !rsGroupMap.containsKey(groupName)) {
246       throw new DoNotRetryIOException("Group "+groupName+" does not exist");
247     }
248 
249     Map<String,RSGroupInfo> newGroupMap = Maps.newHashMap(rsGroupMap);
250     for(TableName tableName: tableNames) {
251       if (tableMap.containsKey(tableName)) {
252         RSGroupInfo src = new RSGroupInfo(newGroupMap.get(tableMap.get(tableName)));
253         src.removeTable(tableName);
254         newGroupMap.put(src.getName(), src);
255       }
256       if(groupName != null) {
257         RSGroupInfo dst = new RSGroupInfo(newGroupMap.get(groupName));
258         dst.addTable(tableName);
259         newGroupMap.put(dst.getName(), dst);
260       }
261     }
262 
263     flushConfig(newGroupMap);
264   }

299   private synchronized void refresh(boolean forceOnline) throws IOException {
300     List<RSGroupInfo> groupList = new LinkedList<RSGroupInfo>();
301 
302     // overwrite anything read from zk, group table is source of truth
303     // if online read from GROUP table
304     if (forceOnline || isOnline()) {
305       LOG.debug("Refreshing in Online mode.");
306       try (Table rsGroupTable = conn.getTable(RSGROUP_TABLE_NAME)) {
307         groupList.addAll(rsGroupSerDe.retrieveGroupList(rsGroupTable));
308       }
309     } else {
310       LOG.debug("Refreshing in Offline mode.");
311       String groupBasePath = ZKUtil.joinZNode(watcher.baseZNode, rsGroupZNode);
312       groupList.addAll(rsGroupSerDe.retrieveGroupList(watcher, groupBasePath));
313     }
314 
315     // refresh default group, prune
316     NavigableSet<TableName> orphanTables = new TreeSet<TableName>();
317     for(String entry: master.getTableDescriptors().getAll().keySet()) {
318       orphanTables.add(TableName.valueOf(entry));
319     }
320 
321     for (RSGroupInfo group: groupList) {
322       if(!group.getName().equals(RSGroupInfo.DEFAULT_GROUP)) {
323         orphanTables.removeAll(group.getTables());
324       }
325     }
326 
327     // This is added to the last of the list
328     // so it overwrites the default group loaded
329     // from region group table or zk
330     groupList.add(new RSGroupInfo(RSGroupInfo.DEFAULT_GROUP,
331         Sets.newHashSet(getDefaultServers()),
332         orphanTables));
333 
334     // populate the data
335     HashMap<String, RSGroupInfo> newGroupMap = Maps.newHashMap();
336     HashMap<TableName, String> newTableMap = Maps.newHashMap();
337     for (RSGroupInfo group : groupList) {
338       newGroupMap.put(group.getName(), group);
339       for(TableName table: group.getTables()) {
340         newTableMap.put(table, group.getName());
341       }
342     }
343     rsGroupMap = Collections.unmodifiableMap(newGroupMap);
344     tableMap = Collections.unmodifiableMap(newTableMap);
345 
346     prevRSGroups.clear();
347     prevRSGroups.addAll(rsGroupMap.keySet());
348   }

350   private synchronized Map<TableName,String> flushConfigTable(Map<String,RSGroupInfo> newGroupMap)
351       throws IOException {
352     Map<TableName,String> newTableMap = Maps.newHashMap();
353     List<Mutation> mutations = Lists.newArrayList();
354 
355     // populate deletes
356     for(String groupName : prevRSGroups) {
357       if(!newGroupMap.containsKey(groupName)) {
358         Delete d = new Delete(Bytes.toBytes(groupName));
359         mutations.add(d);
360       }
361     }
362 
363     // populate puts
364     for(RSGroupInfo RSGroupInfo : newGroupMap.values()) {
365       RSGroupProtos.RSGroupInfo proto = RSGroupProtobufUtil.toProtoGroupInfo(RSGroupInfo);
366       Put p = new Put(Bytes.toBytes(RSGroupInfo.getName()));
367       p.addColumn(META_FAMILY_BYTES,
368           META_QUALIFIER_BYTES,
369           proto.toByteArray());
370       mutations.add(p);
371       for(TableName entry: RSGroupInfo.getTables()) {
372         newTableMap.put(entry, RSGroupInfo.getName());
373       }
374     }
375 
376     if(mutations.size() > 0) {
377       multiMutate(mutations);
378     }
379     return newTableMap;
380   }

382   private synchronized void flushConfig(Map<String, RSGroupInfo> newGroupMap) throws IOException {
383     Map<TableName, String> newTableMap;
384 
385     // For offline mode persistence is still unavailable
386     // We're refreshing in-memory state but only for default servers
387     if (!isOnline()) {
388       Map<String, RSGroupInfo> m = Maps.newHashMap(rsGroupMap);
389       RSGroupInfo oldDefaultGroup = m.remove(RSGroupInfo.DEFAULT_GROUP);
390       RSGroupInfo newDefaultGroup = newGroupMap.remove(RSGroupInfo.DEFAULT_GROUP);
391       if (!m.equals(newGroupMap) ||
392           !oldDefaultGroup.getTables().equals(newDefaultGroup.getTables())) {
393         throw new IOException("Only default servers can be updated during offline mode");
394       }
395       newGroupMap.put(RSGroupInfo.DEFAULT_GROUP, newDefaultGroup);
396       rsGroupMap = newGroupMap;
397       return;
398     }
399 
400     newTableMap = flushConfigTable(newGroupMap);
401 
402     // make changes visible since it has been
403     // persisted in the source of truth
404     rsGroupMap = Collections.unmodifiableMap(newGroupMap);
405     tableMap = Collections.unmodifiableMap(newTableMap);
406 
407 
408     try {
409       String groupBasePath = ZKUtil.joinZNode(watcher.baseZNode, rsGroupZNode);
410       ZKUtil.createAndFailSilent(watcher, groupBasePath, ProtobufUtil.PB_MAGIC);
411 
412       List<ZKUtil.ZKUtilOp> zkOps = new ArrayList<ZKUtil.ZKUtilOp>(newGroupMap.size());
413       for(String groupName : prevRSGroups) {
414         if(!newGroupMap.containsKey(groupName)) {
415           String znode = ZKUtil.joinZNode(groupBasePath, groupName);
416           zkOps.add(ZKUtil.ZKUtilOp.deleteNodeFailSilent(znode));
417         }
418       }
419 
420 
421       for(RSGroupInfo RSGroupInfo : newGroupMap.values()) {
422         String znode = ZKUtil.joinZNode(groupBasePath, RSGroupInfo.getName());
423         RSGroupProtos.RSGroupInfo proto = RSGroupProtobufUtil.toProtoGroupInfo(RSGroupInfo);
424         LOG.debug("Updating znode: "+znode);
425         ZKUtil.createAndFailSilent(watcher, znode);
426         zkOps.add(ZKUtil.ZKUtilOp.deleteNodeFailSilent(znode));
427         zkOps.add(ZKUtil.ZKUtilOp.createAndFailSilent(znode,
428             ProtobufUtil.prependPBMagic(proto.toByteArray())));
429       }
430       LOG.debug("Writing ZK GroupInfo count: " + zkOps.size());
431 
432       ZKUtil.multiOrSequential(watcher, zkOps, false);
433     } catch (KeeperException e) {
434       LOG.error("Failed to write to rsGroupZNode", e);
435       master.abort("Failed to write to rsGroupZNode", e);
436       throw new IOException("Failed to write to rsGroupZNode",e);
437     }
438 
439     prevRSGroups.clear();
440     prevRSGroups.addAll(newGroupMap.keySet());
441   }

443   private List<ServerName> getOnlineRS() throws IOException {
444     if (master != null) {

448       LOG.debug("Reading online RS from zookeeper");

459   private List<Address> getDefaultServers() throws IOException {
460     List<Address> defaultServers = new LinkedList<Address>();
461     for(ServerName server : getOnlineRS()) {
462       Address address = Address.fromParts(server.getHostname(), server.getPort());
463       boolean found = false;
464       for(RSGroupInfo info : rsGroupMap.values()) {
465         if(!RSGroupInfo.DEFAULT_GROUP.equals(info.getName()) &&
466             info.containsServer(address)) {
467           found = true;
468           break;
469         }
470       }
471       if(!found) {
472         defaultServers.add(address);
473       }
474     }
475     return defaultServers;
476   }

498   private static class DefaultServerUpdater extends Thread {
499     private static final Log LOG = LogFactory.getLog(DefaultServerUpdater.class);
500     private RSGroupInfoManagerImpl mgr;
501     private volatile boolean hasChanged = false;
502 
503     public DefaultServerUpdater(RSGroupInfoManagerImpl mgr) {
504       this.mgr = mgr;
505       setName(DefaultServerUpdater.class.getName()+"-" + mgr.master.getServerName());
506       setDaemon(true);
507     }
508 
509     @Override
510     public void run() {
511       List<Address> prevDefaultServers = new LinkedList<Address>();
512       while (!mgr.master.isAborted() && !mgr.master.isStopped()) {
513         try {
514           if (LOG.isDebugEnabled()) {
515             LOG.debug("Updating default servers");
516           }
517           List<Address> servers = mgr.getDefaultServers();
518           Collections.sort(servers, new Comparator<Address>() {
519             @Override
520             public int compare(Address o1, Address o2) {
521               int diff = o1.getHostname().compareTo(o2.getHostname());
522               if (diff != 0) {
523                 return diff;
524               }
525               return o1.getPort() - o2.getPort();
526             }
527           });
528           if(!servers.equals(prevDefaultServers)) {
529             mgr.updateDefaultServers(Sets.<Address>newHashSet(servers));
530             prevDefaultServers = servers;
531             LOG.info("Updated with servers: "+servers.size());
532           }
533           try {
534             synchronized (this) {
535               while (!hasChanged) {
536                 wait();
537               }
538               hasChanged = false;
539             }
540           } catch (InterruptedException e) {
541             LOG.warn("Interrupted", e);
542           }
543         } catch (IOException e) {
544           LOG.warn("Failed to update default servers", e);
545         }
546       }
547     }
548 
549     // Called for both server additions and removals
550     public void serverChanged() {
551       synchronized (this) {
552         hasChanged = true;
553         this.notify();
554       }
555     }
556   }

558   private static class FailedOpenUpdater extends Thread {
559     private static final Log LOG = LogFactory.getLog(FailedOpenUpdater.class);
560 
561     private final RSGroupInfoManagerImpl mgr;
562     private final long waitInterval;
563     private volatile boolean hasChanged = false;
564 
565     public FailedOpenUpdater(RSGroupInfoManagerImpl mgr) {
566       this.mgr = mgr;
567       this.waitInterval = mgr.master.getConfiguration().getLong(REASSIGN_WAIT_INTERVAL_KEY,
568         DEFAULT_REASSIGN_WAIT_INTERVAL);
569       setName(FailedOpenUpdater.class.getName()+"-" + mgr.master.getServerName());
570       setDaemon(true);
571     }
572 
573     @Override
574     public void run() {
575       while (!mgr.master.isAborted() && !mgr.master.isStopped()) {
576         boolean interrupted = false;
577         try {
578           synchronized (this) {
579             while (!hasChanged) {
580               wait();
581             }
582             hasChanged = false;
583           }
584         } catch (InterruptedException e) {
585           LOG.warn("Interrupted", e);
586           interrupted = true;
587         }
588         if (mgr.master.isAborted() || mgr.master.isStopped() || interrupted) {
589           continue;
590         }
591 
592         // First, wait a while in case more servers are about to rejoin the cluster
593         try {
594           Thread.sleep(waitInterval);
595         } catch (InterruptedException e) {
596           LOG.warn("Interrupted", e);
597         }
598         if (mgr.master.isAborted() || mgr.master.isStopped()) {
599           continue;
600         }
601 
602         // Kick all regions in FAILED_OPEN state
603         List<HRegionInfo> failedAssignments = Lists.newArrayList();
604         for (RegionState state:
605             mgr.master.getAssignmentManager().getRegionStates().getRegionsInTransition()) {
606           if (state.isFailedOpen()) {
607             failedAssignments.add(state.getRegion());
608           }
609         }
610         for (HRegionInfo region: failedAssignments) {
611           LOG.info("Retrying assignment of " + region);
612           mgr.master.getAssignmentManager().unassign(region);
613         }
614       }
615     }
616 
617     // Only called for server additions
618     public void serverChanged() {
619       synchronized (this) {
620         hasChanged = true;
621         this.notify();
622       }
623     }
624   }

656     public boolean waitForGroupTableOnline() {
657       final List<HRegionInfo> foundRegions = new LinkedList<HRegionInfo>();
658       final List<HRegionInfo> assignedRegions = new LinkedList<HRegionInfo>();
659       final AtomicBoolean found = new AtomicBoolean(false);
660       final TableStateManager tsm =
661           masterServices.getAssignmentManager().getTableStateManager();
662       boolean createSent = false;
663       while (!found.get() && isMasterRunning()) {
664         foundRegions.clear();
665         assignedRegions.clear();
666         found.set(true);
667         try {
668           boolean rootMetaFound =
669               masterServices.getMetaTableLocator().verifyMetaRegionLocation(
670                   conn,
671                   masterServices.getZooKeeper(),
672                   1);
673           final AtomicBoolean nsFound = new AtomicBoolean(false);
674           if (rootMetaFound) {
675 
676             MetaTableAccessor.Visitor visitor = new MetaTableAccessor.Visitor() {
677               @Override
678               public boolean visit(Result row) throws IOException {
679 
680                 HRegionInfo info = MetaTableAccessor.getHRegionInfo(row);
681                 if (info != null) {
682                   Cell serverCell =
683                       row.getColumnLatestCell(HConstants.CATALOG_FAMILY,
684                           HConstants.SERVER_QUALIFIER);
685                   if (RSGROUP_TABLE_NAME.equals(info.getTable()) && serverCell != null) {
686                     ServerName sn =
687                         ServerName.parseVersionedServerName(CellUtil.cloneValue(serverCell));
688                     if (sn == null) {
689                       found.set(false);
690                     } else if (tsm.isTableState(RSGROUP_TABLE_NAME,
691                         ZooKeeperProtos.Table.State.ENABLED)) {
692                       try {
693                         ClientProtos.ClientService.BlockingInterface rs = conn.getClient(sn);
694                         ClientProtos.GetRequest request =
695                             RequestConverter.buildGetRequest(info.getRegionName(),
696                                 new Get(ROW_KEY));
697                         rs.get(null, request);
698                         assignedRegions.add(info);
699                       } catch(Exception ex) {
700                         LOG.debug("Caught exception while verifying group region", ex);
701                       }
702                     }
703                     foundRegions.add(info);
704                   }
705                   if (TableName.NAMESPACE_TABLE_NAME.equals(info.getTable())) {
706                     Cell cell = row.getColumnLatestCell(HConstants.CATALOG_FAMILY,
707                         HConstants.SERVER_QUALIFIER);
708                     ServerName sn = null;
709                     if(cell != null) {
710                       sn = ServerName.parseVersionedServerName(CellUtil.cloneValue(cell));
711                     }
712                     if (sn == null) {
713                       nsFound.set(false);
714                     } else if (tsm.isTableState(TableName.NAMESPACE_TABLE_NAME,
715                         ZooKeeperProtos.Table.State.ENABLED)) {
716                       try {
717                         ClientProtos.ClientService.BlockingInterface rs = conn.getClient(sn);
718                         ClientProtos.GetRequest request =
719                             RequestConverter.buildGetRequest(info.getRegionName(),
720                                 new Get(ROW_KEY));
721                         rs.get(null, request);
722                         nsFound.set(true);
723                       } catch(Exception ex) {
724                         LOG.debug("Caught exception while verifying group region", ex);
725                       }
726                     }
727                   }
728                 }
729                 return true;
730               }
731             };
732             MetaTableAccessor.fullScan(conn, visitor);
733             // if no regions in meta then we have to create the table
734             if (foundRegions.size() < 1 && rootMetaFound && !createSent && nsFound.get()) {
735               groupInfoManager.createGroupTable(masterServices);
736               createSent = true;
737             }
738             LOG.info("Group table: " + RSGROUP_TABLE_NAME + " isOnline: " + found.get()
739                 + ", regionCount: " + foundRegions.size() + ", assignCount: "
740                 + assignedRegions.size() + ", rootMetaFound: "+rootMetaFound);
741             found.set(found.get() && assignedRegions.size() == foundRegions.size()
742                 && foundRegions.size() > 0);
743           } else {
744             LOG.info("Waiting for catalog tables to come online");
745             found.set(false);
746           }
747           if (found.get()) {
748             LOG.debug("With group table online, refreshing cached information.");
749             groupInfoManager.refresh(true);
750             isOnline = true;
751             //flush any inconsistencies between ZK and HTable
752             groupInfoManager.flushConfig(groupInfoManager.rsGroupMap);
753           }
754         } catch (RuntimeException e) {
755           throw e;
756         } catch(Exception e) {
757           found.set(false);
758           LOG.warn("Failed to perform check", e);
759         }
760         try {
761           Thread.sleep(100);
762         } catch (InterruptedException e) {
763           LOG.info("Sleep interrupted", e);
764         }
765       }
766       return found.get();
767     }

785             RSGROUP_TABLE_DESC,
786             newRegions,

867   @Override
868   public synchronized void removeServers(Set<Address> servers) throws IOException {
869     Map<String, RSGroupInfo> rsGroupInfos = new HashMap<String, RSGroupInfo>();
870     for (Address el: servers) {
871       RSGroupInfo rsGroupInfo = getRSGroupOfServer(el);
872       if (rsGroupInfo != null) {
873         RSGroupInfo newRsGroupInfo = rsGroupInfos.get(rsGroupInfo.getName());
874         if (newRsGroupInfo == null) {
875           rsGroupInfo.removeServer(el);
876           rsGroupInfos.put(rsGroupInfo.getName(), rsGroupInfo);
877         } else {
878           newRsGroupInfo.removeServer(el);
879           rsGroupInfos.put(newRsGroupInfo.getName(), newRsGroupInfo);
880         }
881       } else {
882         LOG.warn("Server " + el + " does not belong to any rsgroup.");
883       }
884     }
885     if (rsGroupInfos.size() > 0) {
886       Map<String, RSGroupInfo> newGroupMap = Maps.newHashMap(rsGroupMap);
887       newGroupMap.putAll(rsGroupInfos);
888       flushConfig(newGroupMap);
889     }
