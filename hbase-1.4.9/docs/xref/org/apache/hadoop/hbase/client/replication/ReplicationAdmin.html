59  /**
60   * <p>
61   * This class provides the administrative interface to HBase cluster
62   * replication. In order to use it, the cluster and the client using
63   * ReplicationAdmin must be configured with <code>hbase.replication</code>
64   * set to true.
65   * </p>
66   * <p>
67   * Adding a new peer results in creating new outbound connections from every
68   * region server to a subset of region servers on the slave cluster. Each
69   * new stream of replication will start replicating from the beginning of the
70   * current WAL, meaning that edits from that past will be replicated.
71   * </p>
72   * <p>
73   * Removing a peer is a destructive and irreversible operation that stops
74   * all the replication streams for the given cluster and deletes the metadata
75   * used to keep track of the replication state.
76   * </p>
77   * <p>
78   * To see which commands are available in the shell, type
79   * <code>replication</code>.
80   * </p>
81   */

108   /**
109    * Constructor that creates a connection to the local ZooKeeper ensemble.
110    * @param conf Configuration to use
111    * @throws IOException if an internal replication error occurs
112    * @throws RuntimeException if replication isn't enabled.
113    */
114   public ReplicationAdmin(Configuration conf) throws IOException {
115     if (!conf.getBoolean(HConstants.REPLICATION_ENABLE_KEY,
116         HConstants.REPLICATION_ENABLE_DEFAULT)) {
117       throw new RuntimeException("hbase.replication isn't true, please " +
118           "enable it in order to use replication");
119     }
120     this.connection = ConnectionFactory.createConnection(conf);
121     try {
122       zkw = createZooKeeperWatcher();
123       try {
124         this.replicationQueuesClient =
125             ReplicationFactory.getReplicationQueuesClient(zkw, conf, this.connection);
126         this.replicationQueuesClient.init();
127         this.replicationPeers = ReplicationFactory.getReplicationPeers(zkw, conf,
128           this.replicationQueuesClient, this.connection);
129         this.replicationPeers.init();
130       } catch (Exception exception) {
131         if (zkw != null) {
132           zkw.close();
133         }
134         throw exception;
135       }
136     } catch (Exception exception) {
137       if (connection != null) {
138         connection.close();
139       }
140       if (exception instanceof IOException) {
141         throw (IOException) exception;
142       } else if (exception instanceof RuntimeException) {
143         throw (RuntimeException) exception;
144       } else {
145         throw new IOException("Error initializing the replication admin client.", exception);
146       }
147     }
148   }

150   private ZooKeeperWatcher createZooKeeperWatcher() throws IOException {
151     // This Abortable doesn't 'abort'... it just logs.
152     return new ZooKeeperWatcher(connection.getConfiguration(), "ReplicationAdmin", new Abortable() {
153       @Override
154       public void abort(String why, Throwable e) {
155         LOG.error(why, e);
156         // We used to call system.exit here but this script can be embedded by other programs that
157         // want to do replication stuff... so inappropriate calling System.exit. Just log for now.
158       }
159 
160       @Override
161       public boolean isAborted() {
162         return false;
163       }
164     });
165   }

226   @VisibleForTesting
227   static String getTableCfsStr(Map<TableName, ? extends Collection<String>> tableCfs) {
228     String tableCfsStr = null;
229     if (tableCfs != null) {
230       // Format: table1:cf1,cf2;table2:cfA,cfB;table3
231       StringBuilder builder = new StringBuilder();
232       for (Entry<TableName, ? extends Collection<String>> entry : tableCfs.entrySet()) {
233         if (builder.length() > 0) {
234           builder.append(";");
235         }
236         builder.append(entry.getKey());
237         if (entry.getValue() != null && !entry.getValue().isEmpty()) {
238           builder.append(":");
239           builder.append(StringUtils.join(entry.getValue(), ","));
240         }
241       }
242       tableCfsStr = builder.toString();
243     }
244     return tableCfsStr;
245   }

330   /**
331    * Append the replicable table-cf config of the specified peer
332    * @param id a short that identifies the cluster
333    * @param tableCfs A map from tableName to column family names
334    */
335   public void appendPeerTableCFs(String id, Map<TableName, ? extends Collection<String>> tableCfs)
336       throws ReplicationException {
337     if (tableCfs == null) {
338       throw new ReplicationException("tableCfs is null");
339     }
340     Map<TableName, List<String>> preTableCfs = this.replicationPeers.getPeerTableCFsConfig(id);
341     if (preTableCfs == null) {
342       setPeerTableCFs(id, tableCfs);
343       return;
344     }
345 
346     for (Map.Entry<TableName, ? extends Collection<String>> entry : tableCfs.entrySet()) {
347       TableName table = entry.getKey();
348       Collection<String> appendCfs = entry.getValue();
349       if (preTableCfs.containsKey(table)) {
350         List<String> cfs = preTableCfs.get(table);
351         if (cfs == null || appendCfs == null) {
352           preTableCfs.put(table, null);
353         } else {
354           Set<String> cfSet = new HashSet<String>(cfs);
355           cfSet.addAll(appendCfs);
356           preTableCfs.put(table, Lists.newArrayList(cfSet));
357         }
358       } else {
359         if (appendCfs == null || appendCfs.isEmpty()) {
360           preTableCfs.put(table, null);
361         } else {
362           preTableCfs.put(table, Lists.newArrayList(appendCfs));
363         }
364       }
365     }
366     setPeerTableCFs(id, preTableCfs);
367   }

379   /**
380    * Remove some table-cfs from config of the specified peer
381    * @param id a short name that identifies the cluster
382    * @param tableCfs A map from tableName to column family names
383    * @throws ReplicationException
384    */
385   public void removePeerTableCFs(String id, Map<TableName, ? extends Collection<String>> tableCfs)
386       throws ReplicationException {
387     if (tableCfs == null) {
388       throw new ReplicationException("tableCfs is null");
389     }
390 
391     Map<TableName, List<String>> preTableCfs = this.replicationPeers.getPeerTableCFsConfig(id);
392     if (preTableCfs == null) {
393       throw new ReplicationException("Table-Cfs for peer" + id + " is null");
394     }
395     for (Map.Entry<TableName, ? extends Collection<String>> entry: tableCfs.entrySet()) {
396       TableName table = entry.getKey();
397       Collection<String> removeCfs = entry.getValue();
398       if (preTableCfs.containsKey(table)) {
399         List<String> cfs = preTableCfs.get(table);
400         if (cfs == null && removeCfs == null) {
401           preTableCfs.remove(table);
402         } else if (cfs != null && removeCfs != null) {
403           Set<String> cfSet = new HashSet<String>(cfs);
404           cfSet.removeAll(removeCfs);
405           if (cfSet.isEmpty()) {
406             preTableCfs.remove(table);
407           } else {
408             preTableCfs.put(table, Lists.newArrayList(cfSet));
409           }
410         } else if (cfs == null && removeCfs != null) {
411           throw new ReplicationException("Cannot remove cf of table: " + table
412               + " which doesn't specify cfs from table-cfs config in peer: " + id);
413         } else if (cfs != null && removeCfs == null) {
414           throw new ReplicationException("Cannot remove table: " + table
415               + " which has specified cfs from table-cfs config in peer: " + id);
416         }
417       } else {
418         throw new ReplicationException("No table: " + table + " in table-cfs config of peer: " + id);
419       }
420     }
421     setPeerTableCFs(id, preTableCfs);
422   }

458   /**
459    * Find all column families that are replicated from this cluster
460    * @return the full list of the replicated column families of this cluster as:
461    *        tableName, family name, replicationType
462    *
463    * Currently replicationType is Global. In the future, more replication
464    * types may be extended here. For example
465    *  1) the replication may only apply to selected peers instead of all peers
466    *  2) the replicationType may indicate the host Cluster servers as Slave
467    *     for the table:columnFam.
468    */
469   public List<HashMap<String, String>> listReplicated() throws IOException {
470     List<HashMap<String, String>> replicationColFams = new ArrayList<HashMap<String, String>>();
471 
472     Admin admin = connection.getAdmin();
473     HTableDescriptor[] tables;
474     try {
475       tables = admin.listTables();
476     } finally {
477       if (admin!= null) admin.close();
478     }
479 
480     for (HTableDescriptor table : tables) {
481       HColumnDescriptor[] columns = table.getColumnFamilies();
482       String tableName = table.getNameAsString();
483       for (HColumnDescriptor column : columns) {
484         if (column.getScope() != HConstants.REPLICATION_SCOPE_LOCAL) {
485           // At this moment, the columfam is replicated to all peers
486           HashMap<String, String> replicationEntry = new HashMap<String, String>();
487           replicationEntry.put(TNAME, tableName);
488           replicationEntry.put(CFNAME, column.getNameAsString());
489           replicationEntry.put(REPLICATIONTYPE, REPLICATIONGLOBAL);
490           replicationColFams.add(replicationEntry);
491         }
492       }
493     }
494 
495     return replicationColFams;
496   }

498   /**
499    * Enable a table's replication switch.
500    * @param tableName name of the table
501    * @throws IOException if a remote or network exception occurs
502    */
503   public void enableTableRep(final TableName tableName) throws IOException {
504     if (tableName == null) {
505       throw new IllegalArgumentException("Table name cannot be null");
506     }
507     try (Admin admin = this.connection.getAdmin()) {
508       if (!admin.tableExists(tableName)) {
509         throw new TableNotFoundException("Table '" + tableName.getNameAsString()
510             + "' does not exists.");
511       }
512     }
513     byte[][] splits = getTableSplitRowKeys(tableName);
514     checkAndSyncTableDescToPeers(tableName, splits);
515     setTableRep(tableName, true);
516   }

531       }
532     }
533     setTableRep(tableName, false);

536   /**
537    * Get the split row keys of table
538    * @param tableName table name
539    * @return array of split row keys
540    * @throws IOException
541    */
542   private byte[][] getTableSplitRowKeys(TableName tableName) throws IOException {
543     try (RegionLocator locator = connection.getRegionLocator(tableName);) {
544       byte[][] startKeys = locator.getStartKeys();
545       if (startKeys.length == 1) {
546         return null;
547       }
548       byte[][] splits = new byte[startKeys.length - 1][];
549       for (int i = 1; i < startKeys.length; i++) {
550         splits[i - 1] = startKeys[i];
551       }
552       return splits;
553     }
554   }

556   /**
557    * Connect to peer and check the table descriptor on peer:
558    * <ol>
559    * <li>Create the same table on peer when not exist.</li>
560    * <li>Throw an exception if the table already has replication enabled on any of the column
561    * families.</li>
562    * <li>Throw an exception if the table exists on peer cluster but descriptors are not same.</li>
563    * </ol>
564    * @param tableName name of the table to sync to the peer
565    * @param splits table split keys
566    * @throws IOException
567    */
568   private void checkAndSyncTableDescToPeers(final TableName tableName, final byte[][] splits)
569       throws IOException {
570     List<ReplicationPeer> repPeers = listReplicationPeers();
571     if (repPeers == null || repPeers.size() <= 0) {
572       throw new IllegalArgumentException("Found no peer cluster for replication.");
573     }
574     
575     final TableName onlyTableNameQualifier = TableName.valueOf(tableName.getQualifierAsString());
576     
577     for (ReplicationPeer repPeer : repPeers) {
578       Map<TableName, List<String>> tableCFMap = repPeer.getTableCFs();
579       // TODO Currently peer TableCFs will not include namespace so we need to check only for table
580       // name without namespace in it. Need to correct this logic once we fix HBASE-11386.
581       if (tableCFMap != null && !tableCFMap.containsKey(onlyTableNameQualifier)) {
582         continue;
583       }
584 
585       Configuration peerConf = repPeer.getConfiguration();
586       HTableDescriptor localHtd = null;
587       try (Connection conn = ConnectionFactory.createConnection(peerConf);
588           Admin admin = this.connection.getAdmin();
589           Admin repHBaseAdmin = conn.getAdmin()) {
590         localHtd = admin.getTableDescriptor(tableName);
591         HTableDescriptor peerHtd = null;
592         if (!repHBaseAdmin.tableExists(tableName)) {
593           repHBaseAdmin.createTable(localHtd, splits);
594         } else {
595           peerHtd = repHBaseAdmin.getTableDescriptor(tableName);
596           if (peerHtd == null) {
597             throw new IllegalArgumentException("Failed to get table descriptor for table "
598                 + tableName.getNameAsString() + " from peer cluster " + repPeer.getId());
599           }
600           if (!compareForReplication(peerHtd, localHtd)) {
601             throw new IllegalArgumentException("Table " + tableName.getNameAsString()
602                 + " exists in peer cluster " + repPeer.getId()
603                 + ", but the table descriptors are not same when compared with source cluster."
604                 + " Thus can not enable the table's replication switch.");
605           }
606         }
607       }
608     }
609   }

616   @VisibleForTesting
617   List<ReplicationPeer> listReplicationPeers() {
618     Map<String, ReplicationPeerConfig> peers = listPeerConfigs();
619     if (peers == null || peers.size() <= 0) {
620       return null;
621     }
622     List<ReplicationPeer> listOfPeers = new ArrayList<ReplicationPeer>(peers.size());
623     for (Entry<String, ReplicationPeerConfig> peerEntry : peers.entrySet()) {
624       String peerId = peerEntry.getKey();
625       try {
626         Pair<ReplicationPeerConfig, Configuration> pair = this.replicationPeers.getPeerConf(peerId);
627         Configuration peerConf = pair.getSecond();
628         ReplicationPeer peer = new ReplicationPeerZKImpl(zkw, pair.getSecond(),
629           peerId, pair.getFirst(), this.connection);
630         listOfPeers.add(peer);
631       } catch (ReplicationException e) {

633             + "Error connecting to peer cluster with peerId=" + peerId + ". Error message="
634             + e.getMessage());
635         LOG.debug("Failure details to get valid replication peers.", e);
636         continue;
637       }

642   /**
643    * Set the table's replication switch if the table's replication switch is already not set.
644    * @param tableName name of the table
645    * @param enableRep is replication switch enable or disable
646    * @throws IOException if a remote or network exception occurs
647    */
648   private void setTableRep(final TableName tableName, boolean enableRep) throws IOException {
649     Admin admin = null;
650     try {
651       admin = this.connection.getAdmin();
652       HTableDescriptor htd = admin.getTableDescriptor(tableName);
653       ReplicationState currentReplicationState = getTableReplicationState(htd);
654       if ((enableRep && currentReplicationState != ReplicationState.ENABLED)
655           || (!enableRep && currentReplicationState != ReplicationState.DISABLED)) {
656         boolean isOnlineSchemaUpdateEnabled =
657             this.connection.getConfiguration()
658                 .getBoolean("hbase.online.schema.update.enable", true);
659         if (!isOnlineSchemaUpdateEnabled) {
660           admin.disableTable(tableName);
661         }
662         for (HColumnDescriptor hcd : htd.getFamilies()) {
663           hcd.setScope(enableRep ? HConstants.REPLICATION_SCOPE_GLOBAL
664               : HConstants.REPLICATION_SCOPE_LOCAL);
665         }
666         admin.modifyTable(tableName, htd);
667         if (!isOnlineSchemaUpdateEnabled) {
668           admin.enableTable(tableName);
669         }
670       }
671     } finally {
672       if (admin != null) {
673         try {
674           admin.close();
675         } catch (IOException e) {
676           LOG.warn("Failed to close admin connection.");
677           LOG.debug("Details on failure to close admin connection.", e);
678         }
679       }
680     }
681   }

692   /**
693    * @param htd table descriptor details for the table to check
694    * @return ReplicationState the current state of the table.
695    */
696   private ReplicationState getTableReplicationState(HTableDescriptor htd) {
697     boolean hasEnabled = false;
698     boolean hasDisabled = false;
699 
700     for (HColumnDescriptor hcd : htd.getFamilies()) {
701       if (hcd.getScope() != HConstants.REPLICATION_SCOPE_GLOBAL) {
702         hasDisabled = true;
703       } else {
704         hasEnabled = true;
705       }
706     }
707 
708     if (hasEnabled && hasDisabled) return ReplicationState.MIXED;
709     if (hasEnabled) return ReplicationState.ENABLED;
710     return ReplicationState.DISABLED;
711   }

732   /**
733    * Copies the REPLICATION_SCOPE of table descriptor passed as an argument. Before copy, the method
734    * ensures that the name of table and column-families should match.
735    * @param peerHtd descriptor on peer cluster
736    * @param localHtd - The HTableDescriptor of table from source cluster.
737    * @return true If the name of table and column families match and REPLICATION_SCOPE copied
738    *         successfully. false If there is any mismatch in the names.
739    */
740   private boolean copyReplicationScope(final HTableDescriptor peerHtd,
741       final HTableDescriptor localHtd) {
742     // Copy the REPLICATION_SCOPE only when table names and the names of
743     // Column-Families are same.
744     int result = peerHtd.getTableName().compareTo(localHtd.getTableName());
745 
746     if (result == 0) {
747       Iterator<HColumnDescriptor> remoteHCDIter = peerHtd.getFamilies().iterator();
748       Iterator<HColumnDescriptor> localHCDIter = localHtd.getFamilies().iterator();
749 
750       while (remoteHCDIter.hasNext() && localHCDIter.hasNext()) {
751         HColumnDescriptor remoteHCD = remoteHCDIter.next();
752         HColumnDescriptor localHCD = localHCDIter.next();
753 
754         String remoteHCDName = remoteHCD.getNameAsString();
755         String localHCDName = localHCD.getNameAsString();
756 
757         if (remoteHCDName.equals(localHCDName)) {
758           remoteHCD.setScope(localHCD.getScope());
759         } else {
760           result = -1;
761           break;
762         }
763       }
764       if (remoteHCDIter.hasNext() || localHCDIter.hasNext()) {
765         return false;
766       }
767     }
768 
769     return result == 0;
770   }

772   /**
773    * Compare the contents of the descriptor with another one passed as a parameter for replication
774    * purpose. The REPLICATION_SCOPE field is ignored during comparison.
775    * @param peerHtd descriptor on peer cluster
776    * @param localHtd descriptor on source cluster which needs to be replicated.
777    * @return true if the contents of the two descriptors match (ignoring just REPLICATION_SCOPE).
778    * @see java.lang.Object#equals(java.lang.Object)
779    */
780   private boolean compareForReplication(HTableDescriptor peerHtd, HTableDescriptor localHtd) {
781     if (peerHtd == null) {
782       return false;
783     }
784     if (peerHtd.equals(localHtd)) {
785       return true;
786     }
787     boolean result = false;
788 
789     // Create a copy of peer HTD as we need to change its replication
790     // scope to match with the local HTD.
791     HTableDescriptor peerHtdCopy = new HTableDescriptor(peerHtd);
792 
793     result = copyReplicationScope(peerHtdCopy, localHtd);
794 
795     // If copy was successful, compare the two tables now.
796     if (result) {
797       result = (peerHtdCopy.compareTo(localHtd) == 0);
798     }
799 
800     return result;
801   }
