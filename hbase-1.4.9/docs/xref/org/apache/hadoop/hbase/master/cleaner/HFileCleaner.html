116   public HFileCleaner(final int period, final Stoppable stopper, Configuration conf, FileSystem fs,
117                       Path directory, Map<String, Object> params) {
118     super("HFileCleaner", period, stopper, conf, fs,
119       directory, MASTER_HFILE_CLEANER_PLUGINS, params);
120     throttlePoint =
121         conf.getInt(HFILE_DELETE_THROTTLE_THRESHOLD, DEFAULT_HFILE_DELETE_THROTTLE_THRESHOLD);
122     largeQueueInitSize =
123         conf.getInt(LARGE_HFILE_QUEUE_INIT_SIZE, DEFAULT_LARGE_HFILE_QUEUE_INIT_SIZE);
124     smallQueueInitSize =
125         conf.getInt(SMALL_HFILE_QUEUE_INIT_SIZE, DEFAULT_SMALL_HFILE_QUEUE_INIT_SIZE);
126     largeFileQueue = new StealJobQueue<>(largeQueueInitSize, smallQueueInitSize);
127     smallFileQueue = largeFileQueue.getStealFromQueue();
128     largeFileDeleteThreadNumber =
129         conf.getInt(LARGE_HFILE_DELETE_THREAD_NUMBER, DEFAULT_LARGE_HFILE_DELETE_THREAD_NUMBER);
130     smallFileDeleteThreadNumber =
131         conf.getInt(SMALL_HFILE_DELETE_THREAD_NUMBER, DEFAULT_SMALL_HFILE_DELETE_THREAD_NUMBER);
132     cleanerThreadTimeoutMsec =
133         conf.getLong(HFILE_DELETE_THREAD_TIMEOUT_MSEC, DEFAULT_HFILE_DELETE_THREAD_TIMEOUT_MSEC);
134     cleanerThreadCheckIntervalMsec =
135         conf.getLong(HFILE_DELETE_THREAD_CHECK_INTERVAL_MSEC,
136             DEFAULT_HFILE_DELETE_THREAD_CHECK_INTERVAL_MSEC);
137     startHFileDeleteThreads();
138   }

155   @Override
156   public int deleteFiles(Iterable<FileStatus> filesToDelete) {
157     int deletedFiles = 0;
158     List<HFileDeleteTask> tasks = new ArrayList<HFileDeleteTask>();
159     // construct delete tasks and add into relative queue
160     for (FileStatus file : filesToDelete) {
161       HFileDeleteTask task = deleteFile(file);
162       if (task != null) {
163         tasks.add(task);
164       }
165     }
166     // wait for each submitted task to finish
167     for (HFileDeleteTask task : tasks) {
168       if (task.getResult(cleanerThreadCheckIntervalMsec)) {
169         deletedFiles++;
170       }
171     }
172     return deletedFiles;
173   }

175   /**
176    * Construct an {@link HFileDeleteTask} for each file to delete and add into the correct queue
177    * @param file the file to delete
178    * @return HFileDeleteTask to track progress
179    */
180   private HFileDeleteTask deleteFile(FileStatus file) {
181     HFileDeleteTask task = new HFileDeleteTask(file, cleanerThreadTimeoutMsec);
182     boolean enqueued = dispatch(task);
183     return enqueued ? task : null;
184   }
185 
186   private boolean dispatch(HFileDeleteTask task) {
187     if (task.fileLength >= this.throttlePoint) {
188       if (!this.largeFileQueue.offer(task)) {
189         // should never arrive here as long as we use PriorityQueue
190         if (LOG.isTraceEnabled()) {
191           LOG.trace("Large file deletion queue is full");
192         }
193         return false;
194       }
195     } else {
196       if (!this.smallFileQueue.offer(task)) {
197         // should never arrive here as long as we use PriorityQueue
198         if (LOG.isTraceEnabled()) {
199           LOG.trace("Small file deletion queue is full");
200         }
201         return false;
202       }
203     }
204     return true;
205   }

213   /**
214    * Start threads for hfile deletion
215    */
216   private void startHFileDeleteThreads() {
217     final String n = Thread.currentThread().getName();
218     running = true;
219     // start thread for large file deletion
220     for (int i = 0; i < largeFileDeleteThreadNumber; i++) {
221       Thread large = new Thread() {
222         @Override
223         public void run() {
224           consumerLoop(largeFileQueue);
225         }
226       };
227       large.setDaemon(true);
228       large.setName(n + "-HFileCleaner.large." + i + "-" + System.currentTimeMillis());
229       large.start();
230       LOG.debug("Starting hfile cleaner for large files: " + large.getName());
231       threads.add(large);
232     }
233 
234     // start thread for small file deletion
235     for (int i = 0; i < smallFileDeleteThreadNumber; i++) {
236       Thread small = new Thread() {
237         @Override
238         public void run() {
239           consumerLoop(smallFileQueue);
240         }
241       };
242       small.setDaemon(true);
243       small.setName(n + "-HFileCleaner.small." + i + "-" + System.currentTimeMillis());
244       small.start();
245       LOG.debug("Starting hfile cleaner for small files: " + small.getName());
246       threads.add(small);
247     }
248   }
249 
250   protected void consumerLoop(BlockingQueue<HFileDeleteTask> queue) {
251     try {
252       while (running) {
253         HFileDeleteTask task = null;
254         try {
255           task = queue.take();
256         } catch (InterruptedException e) {
257           if (LOG.isDebugEnabled()) {
258             LOG.debug("Interrupted while trying to take a task from queue", e);
259           }
260           break;
261         }
262         if (task != null) {
263           if (LOG.isDebugEnabled()) {
264             LOG.debug("Removing: " + task.filePath + " from archive");
265           }
266           boolean succeed;
267           try {
268             succeed = this.fs.delete(task.filePath, false);
269           } catch (IOException e) {
270             LOG.warn("Failed to delete file " + task.filePath, e);
271             succeed = false;
272           }
273           task.setResult(succeed);
274           if (succeed) {
275             countDeletedFiles(task.fileLength >= throttlePoint, queue == largeFileQueue);
276           }
277         }
278       }
279     } finally {
280       if (LOG.isDebugEnabled()) {
281         LOG.debug("Exit thread: " + Thread.currentThread());
282       }
283     }
284   }

286   // Currently only for testing purpose
287   private void countDeletedFiles(boolean isLargeFile, boolean fromLargeQueue) {
288     if (isLargeFile) {
289       if (deletedLargeFiles.get() == Long.MAX_VALUE) {
290         LOG.info("Deleted more than Long.MAX_VALUE large files, reset counter to 0");
291         deletedLargeFiles.set(0L);
292       }
293       deletedLargeFiles.incrementAndGet();
294     } else {
295       if (deletedSmallFiles.get() == Long.MAX_VALUE) {
296         LOG.info("Deleted more than Long.MAX_VALUE small files, reset counter to 0");
297         deletedSmallFiles.set(0L);
298       }
299       if (fromLargeQueue && LOG.isTraceEnabled()) {
300         LOG.trace("Stolen a small file deletion task in large file thread");
301       }
302       deletedSmallFiles.incrementAndGet();
303     }
304   }

306   /**
307    * Stop threads for hfile deletion
308    */
309   private void stopHFileDeleteThreads() {
310     running = false;
311     if (LOG.isDebugEnabled()) {
312       LOG.debug("Stopping file delete threads");
313     }
314     for(Thread thread: threads){
315       thread.interrupt();
316     }

319   static class HFileDeleteTask implements Comparable<HFileDeleteTask> {
320 
321     boolean done = false;
322     boolean result;
323     final Path filePath;
324     final long fileLength;
325     final long timeoutMsec;
326 
327     public HFileDeleteTask(FileStatus file, long timeoutMsec) {
328       this.filePath = file.getPath();
329       this.fileLength = file.getLen();
330       this.timeoutMsec = timeoutMsec;
331     }
332 
333     public synchronized void setResult(boolean result) {
334       this.done = true;
335       this.result = result;
336       notify();
337     }
338 
339     public synchronized boolean getResult(long waitIfNotFinished) {
340       long waitTimeMsec = 0;
341       try {
342         while (!done) {
343           long startTimeNanos = System.nanoTime();
344           wait(waitIfNotFinished);
345           waitTimeMsec += TimeUnit.MILLISECONDS.convert(System.nanoTime() - startTimeNanos,
346               TimeUnit.NANOSECONDS);
347           if (done) {
348             return this.result;
349           }
350           if (waitTimeMsec > timeoutMsec) {
351             LOG.warn("Wait more than " + timeoutMsec + " ms for deleting " + this.filePath
352                 + ", exit...");
353             return false;
354           }
355         }
356       } catch (InterruptedException e) {
357         LOG.warn("Interrupted while waiting for result of deleting " + filePath
358             + ", will return false", e);
359         return false;
360       }
361       return this.result;
362     }
363 
364     @Override
365     public int compareTo(HFileDeleteTask o) {
366       long sub = this.fileLength - o.fileLength;
367       // smaller value with higher priority in PriorityQueue, and we intent to delete the larger
368       // file first.
369       return (sub > 0) ? -1 : (sub < 0 ? 1 : 0);
370     }
371 
372     @Override
373     public boolean equals(Object o) {
374       if (this == o) {
375         return true;
376       }
377       if (o == null || !(o instanceof HFileDeleteTask)) {
378         return false;
379       }
380       HFileDeleteTask otherTask = (HFileDeleteTask) o;
381       return this.filePath.equals(otherTask.filePath) && (this.fileLength == otherTask.fileLength);
382     }
383 
384     @Override
385     public int hashCode() {
386       return filePath.hashCode();
387     }
388   }

430   @Override
431   public void onConfigurationChange(Configuration conf) {
432     if (!checkAndUpdateConfigurations(conf)) {
433       LOG.debug("Update configuration triggered but nothing changed for this cleaner");
434       return;
435     }
436     stopHFileDeleteThreads();
437     // record the left over tasks
438     List<HFileDeleteTask> leftOverTasks =
439         new ArrayList<>(largeFileQueue.size() + smallFileQueue.size());
440     for (HFileDeleteTask task : largeFileQueue) {
441       leftOverTasks.add(task);
442     }
443     for (HFileDeleteTask task : smallFileQueue) {
444       leftOverTasks.add(task);
445     }
446     largeFileQueue = new StealJobQueue<>(largeQueueInitSize, smallQueueInitSize);
447     smallFileQueue = largeFileQueue.getStealFromQueue();
448     threads.clear();
449     startHFileDeleteThreads();
450     // re-dispatch the left over tasks
451     for (HFileDeleteTask task : leftOverTasks) {
452       dispatch(task);
453     }
454   }

456   /**
457    * Check new configuration and update settings if value changed
458    * @param conf The new configuration
459    * @return true if any configuration for HFileCleaner changes, false if no change
460    */
461   private boolean checkAndUpdateConfigurations(Configuration conf) {
462     boolean updated = false;
463     int throttlePoint =
464         conf.getInt(HFILE_DELETE_THROTTLE_THRESHOLD, DEFAULT_HFILE_DELETE_THROTTLE_THRESHOLD);
465     if (throttlePoint != this.throttlePoint) {
466       LOG.debug("Updating throttle point, from " + this.throttlePoint + " to " + throttlePoint);
467       this.throttlePoint = throttlePoint;
468       updated = true;
469     }
470     int largeQueueInitSize =
471         conf.getInt(LARGE_HFILE_QUEUE_INIT_SIZE, DEFAULT_LARGE_HFILE_QUEUE_INIT_SIZE);
472     if (largeQueueInitSize != this.largeQueueInitSize) {
473       LOG.debug("Updating largeQueueInitSize, from " + this.largeQueueInitSize + " to "
474           + largeQueueInitSize);
475       this.largeQueueInitSize = largeQueueInitSize;
476       updated = true;
477     }
478     int smallQueueInitSize =
479         conf.getInt(SMALL_HFILE_QUEUE_INIT_SIZE, DEFAULT_SMALL_HFILE_QUEUE_INIT_SIZE);
480     if (smallQueueInitSize != this.smallQueueInitSize) {
481       LOG.debug("Updating smallQueueInitSize, from " + this.smallQueueInitSize + " to "
482           + smallQueueInitSize);
483       this.smallQueueInitSize = smallQueueInitSize;
484       updated = true;
485     }
486     int largeFileDeleteThreadNumber =
487         conf.getInt(LARGE_HFILE_DELETE_THREAD_NUMBER, DEFAULT_LARGE_HFILE_DELETE_THREAD_NUMBER);
488     if (largeFileDeleteThreadNumber != this.largeFileDeleteThreadNumber) {
489       LOG.debug("Updating largeFileDeleteThreadNumber, from " + this.largeFileDeleteThreadNumber
490           + " to " + largeFileDeleteThreadNumber);
491       this.largeFileDeleteThreadNumber = largeFileDeleteThreadNumber;
492       updated = true;
493     }
494     int smallFileDeleteThreadNumber =
495         conf.getInt(SMALL_HFILE_DELETE_THREAD_NUMBER, DEFAULT_SMALL_HFILE_DELETE_THREAD_NUMBER);
496     if (smallFileDeleteThreadNumber != this.smallFileDeleteThreadNumber) {
497       LOG.debug("Updating smallFileDeleteThreadNumber, from " + this.smallFileDeleteThreadNumber
498           + " to " + smallFileDeleteThreadNumber);
499       this.smallFileDeleteThreadNumber = smallFileDeleteThreadNumber;
500       updated = true;
501     }
502     long cleanerThreadTimeoutMsec =
503         conf.getLong(HFILE_DELETE_THREAD_TIMEOUT_MSEC, DEFAULT_HFILE_DELETE_THREAD_TIMEOUT_MSEC);
504     if (cleanerThreadTimeoutMsec != this.cleanerThreadTimeoutMsec) {
505       this.cleanerThreadTimeoutMsec = cleanerThreadTimeoutMsec;
506       updated = true;
507     }
508     long cleanerThreadCheckIntervalMsec =
509         conf.getLong(HFILE_DELETE_THREAD_CHECK_INTERVAL_MSEC,
510             DEFAULT_HFILE_DELETE_THREAD_CHECK_INTERVAL_MSEC);
511     if (cleanerThreadCheckIntervalMsec != this.cleanerThreadCheckIntervalMsec) {
512       this.cleanerThreadCheckIntervalMsec = cleanerThreadCheckIntervalMsec;
513       updated = true;
514     }
515     return updated;
516   }
