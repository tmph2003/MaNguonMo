128   public RpcExecutor(final String name, final int handlerCount, final String callQueueType,
129       final int maxQueueLength, final PriorityFunction priority, final Configuration conf,
130       final Abortable abortable) {
131     this.name = Strings.nullToEmpty(name);
132     this.conf = conf;
133     this.abortable = abortable;
134 
135     float callQueuesHandlersFactor = this.conf.getFloat(CALL_QUEUE_HANDLER_FACTOR_CONF_KEY, 0);
136     this.numCallQueues = computeNumCallQueues(handlerCount, callQueuesHandlersFactor);
137     this.queues = new ArrayList<>(this.numCallQueues);
138 
139     this.handlerCount = Math.max(handlerCount, this.numCallQueues);
140     this.handlers = new ArrayList<>(this.handlerCount);
141 
142     this.priority = priority;
143 
144     if (isDeadlineQueueType(callQueueType)) {
145       this.name += ".Deadline";
146       this.queueInitArgs = new Object[] { maxQueueLength,
147         new CallPriorityComparator(conf, this.priority) };
148       this.queueClass = BoundedPriorityBlockingQueue.class;
149     } else if (isCodelQueueType(callQueueType)) {
150       this.name += ".Codel";
151       int codelTargetDelay = conf.getInt(CALL_QUEUE_CODEL_TARGET_DELAY,
152         CALL_QUEUE_CODEL_DEFAULT_TARGET_DELAY);
153       int codelInterval = conf.getInt(CALL_QUEUE_CODEL_INTERVAL, CALL_QUEUE_CODEL_DEFAULT_INTERVAL);
154       double codelLifoThreshold = conf.getDouble(CALL_QUEUE_CODEL_LIFO_THRESHOLD,
155         CALL_QUEUE_CODEL_DEFAULT_LIFO_THRESHOLD);
156       queueInitArgs = new Object[] { maxQueueLength, codelTargetDelay, codelInterval,
157           codelLifoThreshold, numGeneralCallsDropped, numLifoModeSwitches };
158       queueClass = AdaptiveLifoCoDelCallQueue.class;
159     } else {
160       this.name += ".Fifo";
161       queueInitArgs = new Object[] { maxQueueLength };
162       queueClass = LinkedBlockingQueue.class;
163     }
164 
165     LOG.info("RpcExecutor " + " name " + " using " + callQueueType
166         + " as call queue; numCallQueues=" + numCallQueues + "; maxQueueLength=" + maxQueueLength
167         + "; handlerCount=" + handlerCount);
168   }

170   protected int computeNumCallQueues(final int handlerCount, final float callQueuesHandlersFactor) {
171     return Math.max(1, (int) Math.round(handlerCount * callQueuesHandlersFactor));
172   }
173 
174   protected void initializeQueues(final int numQueues) {
175     if (queueInitArgs.length > 0) {
176       currentQueueLimit = (int) queueInitArgs[0];
177       queueInitArgs[0] = Math.max((int) queueInitArgs[0], DEFAULT_CALL_QUEUE_SIZE_HARD_LIMIT);
178     }
179     for (int i = 0; i < numQueues; ++i) {
180       queues
181           .add((BlockingQueue<CallRunner>) ReflectionUtils.newInstance(queueClass, queueInitArgs));
182     }
183   }

218   /**
219    * Start up our handlers.
220    */
221   protected void startHandlers(final String nameSuffix, final int numHandlers,
222       final List<BlockingQueue<CallRunner>> callQueues, final int qindex, final int qsize,
223       final int port, final AtomicInteger activeHandlerCount) {
224     final String threadPrefix = name + Strings.nullToEmpty(nameSuffix);
225     double handlerFailureThreshhold = conf == null ? 1.0 : conf.getDouble(
226       HConstants.REGION_SERVER_HANDLER_ABORT_ON_ERROR_PERCENT,
227       HConstants.DEFAULT_REGION_SERVER_HANDLER_ABORT_ON_ERROR_PERCENT);
228     for (int i = 0; i < numHandlers; i++) {
229       final int index = qindex + (i % qsize);
230       String name = "RpcServer." + threadPrefix + ".handler=" + handlers.size() + ",queue=" + index
231           + ",port=" + port;
232       Handler handler = getHandler(name, handlerFailureThreshhold, callQueues.get(index),
233         activeHandlerCount);
234       handler.start();
235       LOG.debug("Started " + name);
236       handlers.add(handler);
237     }

240   /**
241    * Handler thread run the {@link CallRunner#run()} in.
242    */
243   protected class Handler extends Thread {
244     /**
245      * Q to find CallRunners to run in.
246      */
247     final BlockingQueue<CallRunner> q;
248 
249     final double handlerFailureThreshhold;
250 
251     // metrics (shared with other handlers)
252     final AtomicInteger activeHandlerCount;
253 
254     Handler(final String name, final double handlerFailureThreshhold,
255         final BlockingQueue<CallRunner> q, final AtomicInteger activeHandlerCount) {
256       super(name);
257       setDaemon(true);
258       this.q = q;
259       this.handlerFailureThreshhold = handlerFailureThreshhold;
260       this.activeHandlerCount = activeHandlerCount;
261     }
262 
263     /**
264      * @return A {@link CallRunner}
265      * @throws InterruptedException
266      */
267     protected CallRunner getCallRunner() throws InterruptedException {
268       return this.q.take();
269     }
270 
271     @Override
272     public void run() {
273       boolean interrupted = false;
274       try {
275         while (running) {
276           try {
277             run(getCallRunner());
278           } catch (InterruptedException e) {
279             interrupted = true;
280           }
281         }
282       } catch (Exception e) {
283         LOG.warn(e);
284         throw e;
285       } finally {
286         if (interrupted) {
287           Thread.currentThread().interrupt();
288         }
289       }
290     }
291 
292     private void run(CallRunner cr) {
293       MonitoredRPCHandler status = RpcServer.getStatus();
294       cr.setStatus(status);
295       try {
296         this.activeHandlerCount.incrementAndGet();
297         cr.run();
298       } catch (Throwable e) {
299         if (e instanceof Error) {
300           int failedCount = failedHandlerCount.incrementAndGet();
301           if (this.handlerFailureThreshhold >= 0
302               && failedCount > handlerCount * this.handlerFailureThreshhold) {
303             String message = "Number of failed RpcServer handler runs exceeded threshhold "
304                 + this.handlerFailureThreshhold + "; reason: " + StringUtils.stringifyException(e);
305             if (abortable != null) {
306               abortable.abort(message, e);
307             } else {
308               LOG.error("Error but can't abort because abortable is null: "
309                   + StringUtils.stringifyException(e));
310               throw e;
311             }
312           } else {
313             LOG.warn("Handler errors " + StringUtils.stringifyException(e));
314           }
315         } else {
316           LOG.warn("Handler  exception " + StringUtils.stringifyException(e));
317         }
318       } finally {
319         this.activeHandlerCount.decrementAndGet();
320       }
321     }

365   /**
366    * Comparator used by the "normal callQueue" if DEADLINE_CALL_QUEUE_CONF_KEY is set to true. It
367    * uses the calculated "deadline" e.g. to deprioritize long-running job If multiple requests have
368    * the same deadline BoundedPriorityBlockingQueue will order them in FIFO (first-in-first-out)
369    * manner.
370    */
371   private static class CallPriorityComparator implements Comparator<CallRunner> {
372     private final static int DEFAULT_MAX_CALL_DELAY = 5000;
373 
374     private final PriorityFunction priority;
375     private final int maxDelay;
376 
377     public CallPriorityComparator(final Configuration conf, final PriorityFunction priority) {
378       this.priority = priority;
379       this.maxDelay = conf.getInt(QUEUE_MAX_CALL_DELAY_CONF_KEY, DEFAULT_MAX_CALL_DELAY);
380     }
381 
382     @Override
383     public int compare(CallRunner a, CallRunner b) {
384       RpcServer.Call callA = a.getCall();
385       RpcServer.Call callB = b.getCall();
386       long deadlineA = priority.getDeadline(callA.getHeader(), callA.param);
387       long deadlineB = priority.getDeadline(callB.getHeader(), callB.param);
388       deadlineA = callA.timestamp + Math.min(deadlineA, maxDelay);
389       deadlineB = callB.timestamp + Math.min(deadlineB, maxDelay);
390       return Long.compare(deadlineA, deadlineB);
391     }
392   }

467   public void onConfigurationChange(Configuration conf) {
468     // update CoDel Scheduler tunables
469     int codelTargetDelay = conf.getInt(CALL_QUEUE_CODEL_TARGET_DELAY,
470       CALL_QUEUE_CODEL_DEFAULT_TARGET_DELAY);
471     int codelInterval = conf.getInt(CALL_QUEUE_CODEL_INTERVAL, CALL_QUEUE_CODEL_DEFAULT_INTERVAL);
472     double codelLifoThreshold = conf.getDouble(CALL_QUEUE_CODEL_LIFO_THRESHOLD,
473       CALL_QUEUE_CODEL_DEFAULT_LIFO_THRESHOLD);
474 
475     for (BlockingQueue<CallRunner> queue : queues) {
476       if (queue instanceof AdaptiveLifoCoDelCallQueue) {
477         ((AdaptiveLifoCoDelCallQueue) queue).updateTunables(codelTargetDelay, codelInterval,
478           codelLifoThreshold);
479       }
480     }
